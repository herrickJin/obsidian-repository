### 分配策略

```
┌─────────────────────────────────────────────────────────────┐
│                    分配策略                                  │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  小对象 (< 128 bytes):                                       │
│  ├── 从 Bucket 分配                                         │
│  ├── 每个 Bucket 固定大小                                   │
│  ├── 8, 16, 24, 32, 48, 64, 80, 96, 128 bytes              │
│  └── 从 Slot Span 快速分配                                  │
│                                                             │
│  中等对象 (128 bytes - 16 KB):                               │
│  ├── 从 Bucket 分配                                         │
│  ├── 按 16 bytes 对齐                                       │
│  └── 可能跨越多个 Partition Page                            │
│                                                             │
│  大对象 (> 16 KB):                                           │
│  ├── 直接从系统分配                                         │
│  ├── 使用 mmap/VirtualAlloc                                 │
│  └── 独立管理                                                │
│                                                             │
│  直接映射 (> 384 KB):                                        │
│  ├── 完全绕过 PartitionAlloc                                │
│  └── 直接系统调用                                            │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### Miracast (MiraclePtr)

```cpp
// MiraclePtr - 安全智能指针

// 传统裸指针问题:
void UnsafeCode() {
  raw_ptr<int> ptr = new int(42);
  delete ptr;
  // ptr 变成悬垂指针
  *ptr;  // 危险！可能崩溃或被利用
}

// MiraclePtr 解决方案:
void SafeCode() {
  miracle_ptr<int> ptr = new int(42);
  delete ptr;
  // ptr 被标记为无效
  *ptr;  // 安全崩溃，不会被利用
}

// 实现:
// - 维护指针引用计数
// - 释放时标记无效
// - 访问时检查有效性
// - 无效时触发安全崩溃
```

## 🐛 内存泄漏检测

### 常见内存泄漏

```javascript
// 常见内存泄漏模式

// 1. 未清理的事件监听
class Component {
  constructor() {
    window.addEventListener('resize', this.onResize)
  }

  onResize = () => {
    // 处理
  }

  // ❌ 忘记在销毁时移除监听
}

// ✅ 正确做法
class Component {
  constructor() {
    window.addEventListener('resize', this.onResize)
  }

  destroy() {
    window.removeEventListener('resize', this.onResize)
  }
}

// 2. 闭包持有大对象
function createHandler() {
  const largeData = new Array(1000000)
  return () => {
    console.log(largeData.length)  // largeData 无法释放
  }
}

// ✅ 只保留需要的值
function createHandler() {
  const largeData = new Array(1000000)
  const length = largeData.length
  return () => console.log(length)
}

// 3. 未清理的定时器
class Timer {
  start() {
    this.timer = setInterval(() => {
      // 处理
    }, 1000)
  }
  // ❌ 忘记清理
}

// ✅ 正确做法
class Timer {
  stop() {
    clearInterval(this.timer)
  }
}

// 4. DOM 引用
class Cache {
  constructor() {
    this.elements = {}
  }

  add(id, element) {
    this.elements[id] = element
  }
  // ❌ 即使 element 从 DOM 移除，仍被引用
}

// ✅ 使用 WeakMap
const cache = new WeakMap()
cache.set(element, data)  // element 可被 GC
```

### 内存压力处理

```cpp
// Chromium 内存压力处理

// 内存压力级别
enum class MemoryPressureLevel {
  kNone,        // 正常
  kModerate,    // 中等压力
  kCritical,    // 严重压力
};

// 响应策略
void OnMemoryPressure(MemoryPressureLevel level) {
  switch (level) {
    case MemoryPressureLevel::kModerate:
      // 清理缓存
      ClearCaches();
      // 释放非必要资源
      ReleaseNonCriticalResources();
      break;

    case MemoryPressureLevel::kCritical:
      // 积极释放
      ClearAllCaches();
      // 降低渲染质量
      ReduceRenderingQuality();
      // 丢弃后台标签页
      DiscardBackgroundTabs();
      break;
  }
}
```
