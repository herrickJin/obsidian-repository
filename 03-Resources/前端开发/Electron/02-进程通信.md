# Electron 进程通信 (IPC)

## 🎯 学习目标

- 掌握 ipcMain 和 ipcRenderer 的使用
- 理解 contextBridge 的安全通信方式
- 学会使用 preload 脚本暴露 API

---

## 📖 核心概念

### 为什么需要 IPC？

```
主进程和渲染进程是隔离的，不能直接共享数据或调用函数。

渲染进程 (网页)  ←── IPC ──→  主进程 (Node.js)
     ↓                              ↓
   用户界面                      系统资源
   (受限)                       (完整权限)
```

### IPC 通信模式

```
┌─────────────────────────────────────────────────────┐
│                    通信模式                          │
├─────────────────────────────────────────────────────┤
│                                                     │
│  1. 异步请求-响应                                    │
│     Renderer ──invoke()──→ Main                    │
│     Renderer ←──return─── Main                     │
│                                                     │
│  2. 同步请求-响应 (不推荐)                           │
│     Renderer ──sendSync()──→ Main                  │
│     Renderer ←──return────── Main                  │
│                                                     │
│  3. 事件驱动                                         │
│     Renderer ──send()────→ Main                    │
│     Main ──────send()────→ Renderer                │
│                                                     │
│  4. 双向通信 (MessageChannel)                       │
│     Renderer ←──── MessagePort ────→ Main          │
│                                                     │
└─────────────────────────────────────────────────────┘
```

## 🔧 基本用法

### 1. 异步请求-响应 (推荐)

**主进程 (main.js)**

```javascript
const { ipcMain } = require('electron')

// 处理异步请求
ipcMain.handle('get-user-info', async (event, userId) => {
  // 可以执行异步操作
  const userInfo = await fetchUserInfo(userId)
  return userInfo
})

// 处理文件读取
ipcMain.handle('read-file', async (event, filePath) => {
  const fs = require('fs').promises
  try {
    const content = await fs.readFile(filePath, 'utf-8')
    return { success: true, content }
  } catch (error) {
    return { success: false, error: error.message }
  }
})
```

**预加载脚本 (preload.js)**

```javascript
const { contextBridge, ipcRenderer } = require('electron')

contextBridge.exposeInMainWorld('electronAPI', {
  // 获取用户信息
  getUserInfo: (userId) => ipcRenderer.invoke('get-user-info', userId),

  // 读取文件
  readFile: (filePath) => ipcRenderer.invoke('read-file', filePath)
})
```

**渲染进程 (renderer.js)**

```javascript
// 使用 async/await
async function loadUserInfo() {
  const userInfo = await window.electronAPI.getUserInfo(123)
  console.log(userInfo)
}

// 或者使用 Promise
window.electronAPI.readFile('/path/to/file.txt')
  .then(result => {
    if (result.success) {
      console.log(result.content)
    }
  })
```

### 2. 事件驱动通信

**主进程监听事件**

```javascript
const { ipcMain } = require('electron')

// 监听渲染进程发送的消息
ipcMain.on('log-message', (event, message) => {
  console.log('From renderer:', message)
})

// 向渲染进程发送消息
ipcMain.on('request-notification', (event, data) => {
  // 处理请求...
  // 发送响应
  event.reply('notification-response', { status: 'sent' })
})
```

**预加载脚本**

```javascript
const { contextBridge, ipcRenderer } = require('electron')

contextBridge.exposeInMainWorld('electronAPI', {
  // 发送消息（单向）
  logMessage: (message) => ipcRenderer.send('log-message', message),

  // 监听主进程消息
  onNotificationResponse: (callback) => {
    ipcRenderer.on('notification-response', (event, data) => callback(data))
  },

  // 移除监听器
  removeListener: (channel) => {
    ipcRenderer.removeAllListeners(channel)
  }
})
```

**渲染进程使用**

```javascript
// 发送消息
window.electronAPI.logMessage('Hello from renderer!')

// 监听响应
window.electronAPI.onNotificationResponse((data) => {
  console.log('Response:', data)
})
```

### 3. 同步通信 (不推荐)

```javascript
// 主进程
ipcMain.on('sync-request', (event, data) => {
  event.returnValue = { result: 'processed' }
})

// 预加载脚本
contextBridge.exposeInMainWorld('electronAPI', {
  syncRequest: (data) => ipcRenderer.sendSync('sync-request', data)
})

// 渲染进程 (会阻塞!)
const result = window.electronAPI.syncRequest({ test: true })
```

> ⚠️ 警告: 同步通信会阻塞渲染进程，影响用户体验，应尽量避免使用！

## 🔒 安全通信模式

### contextBridge 详解

```javascript
// preload.js
const { contextBridge, ipcRenderer } = require('electron')

// ✅ 正确做法：只暴露必要的、经过验证的 API
contextBridge.exposeInMainWorld('electronAPI', {
  // 使用白名单验证 channel
  send: (channel, data) => {
    const validChannels = ['toMain', 'update-setting']
    if (validChannels.includes(channel)) {
      ipcRenderer.send(channel, data)
    }
  },

  receive: (channel, func) => {
    const validChannels = ['fromMain', 'update-progress']
    if (validChannels.includes(channel)) {
      // 包装函数，移除 event 对象
      ipcRenderer.on(channel, (event, ...args) => func(...args))
    }
  },

  // 使用 invoke (推荐)
  invoke: (channel, data) => {
    const validChannels = ['dialog:open', 'file:read']
    if (validChannels.includes(channel)) {
      return ipcRenderer.invoke(channel, data)
    }
    return Promise.reject('Invalid channel')
  }
})
```

### 完整的安全通信示例

```javascript
// preload.js
const { contextBridge, ipcRenderer } = require('electron')

contextBridge.exposeInMainWorld('electronAPI', {
  // 文件操作
  file: {
    open: () => ipcRenderer.invoke('dialog:openFile'),
    save: (content) => ipcRenderer.invoke('file:save', content),
    read: (path) => ipcRenderer.invoke('file:read', path)
  },

  // 应用信息
  app: {
    getVersion: () => ipcRenderer.invoke('app:getVersion'),
    getPath: (name) => ipcRenderer.invoke('app:getPath', name)
  },

  // 系统通知
  notification: {
    show: (title, body) => ipcRenderer.send('notification:show', { title, body }),
    onClick: (callback) => ipcRenderer.on('notification:clicked', callback)
  }
})

// main.js
const { app, BrowserWindow, ipcMain, dialog } = require('electron')
const fs = require('fs').promises

// 文件对话框
ipcMain.handle('dialog:openFile', async () => {
  const result = await dialog.showOpenDialog({
    filters: [{ name: 'Text', extensions: ['txt', 'md'] }]
  })
  return result.filePaths[0]
})

// 保存文件
ipcMain.handle('file:save', async (event, { path, content }) => {
  await fs.writeFile(path, content)
  return true
})

// 读取文件
ipcMain.handle('file:read', async (event, path) => {
  const content = await fs.readFile(path, 'utf-8')
  return content
})

// 应用版本
ipcMain.handle('app:getVersion', () => app.getVersion())

// 应用路径
ipcMain.handle('app:getPath', (event, name) => app.getPath(name))
```

## 🔧 高级用法

### MessageChannel 双向通信

```javascript
// 主进程
ipcMain.on('create-channel', (event) => {
  // 创建消息端口
  const { port1, port2 } = new MessageChannelMain()

  // 发送一个端口给渲染进程
  event.sender.postMessage('port', null, [port1])

  // 主进程使用另一个端口
  port2.on('message', (event) => {
    console.log('Received:', event.data)
    port2.postMessage('Reply from main')
  })
  port2.start()
})

// 预加载脚本
contextBridge.exposeInMainWorld('electronAPI', {
  createChannel: () => {
    return new Promise((resolve) => {
      ipcRenderer.on('port', (event) => {
        const [port] = event.ports
        resolve({
          send: (data) => port.postMessage(data),
          onMessage: (callback) => {
            port.onmessage = (e) => callback(e.data)
          },
          start: () => port.start()
        })
      })
      ipcRenderer.send('create-channel')
    })
  }
})
```

## 📊 IPC 通信对比

| 方式 | 特点 | 适用场景 | 性能 |
|------|------|----------|------|
| `invoke/handle` | 异步、Promise | 请求-响应 | 优秀 |
| `send/on` | 异步、事件 | 单向通知 | 优秀 |
| `sendSync/on` | 同步、阻塞 | 极少使用 | 差 |
| `MessageChannel` | 双向流 | 高频通信 | 最佳 |
