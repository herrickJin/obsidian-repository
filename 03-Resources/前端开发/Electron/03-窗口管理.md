## 📖 核心概念

### BrowserWindow 基础

```javascript
const { BrowserWindow } = require('electron')

const win = new BrowserWindow({
  // 尺寸
  width: 800,
  height: 600,
  minWidth: 400,
  maxWidth: 1200,
  minHeight: 300,
  maxHeight: 900,

  // 位置
  x: 100,
  y: 100,
  center: true,  // 居中显示

  // 外观
  title: 'My App',
  icon: '/path/to/icon.png',
  frame: true,           // 是否显示边框
  transparent: false,    // 透明窗口
  backgroundColor: '#fff',

  // 行为
  resizable: true,
  movable: true,
  minimizable: true,
  maximizable: true,
  closable: true,
  alwaysOnTop: false,
  fullscreen: false,
  kiosk: false,          // Kiosk 模式

  // Web 相关
  webPreferences: {
    preload: path.join(__dirname, 'preload.js'),
    contextIsolation: true,
    nodeIntegration: false,
    webSecurity: true,
    devTools: true
  }
})
```

### 窗口类型

```
┌─────────────────────────────────────────────────────────────┐
│                       窗口类型                               │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  1. 普通窗口 (Normal Window)                                 │
│     - 标准的桌面应用窗口                                      │
│     - 有标题栏、边框、控制按钮                                │
│                                                             │
│  2. 无边框窗口 (Frameless Window)                            │
│     - 无系统标题栏                                           │
│     - 需要自己实现拖拽、关闭等                                │
│                                                             │
│  3. 透明窗口 (Transparent Window)                            │
│     - 支持透明背景                                           │
│     - 可实现不规则形状                                        │
│                                                             │
│  4. 模态窗口 (Modal Window)                                  │
│     - 父窗口禁用                                             │
│     - 必须先处理模态窗口                                      │
│                                                             │
│  5. 子窗口 (Child Window)                                    │
│     - 相对于父窗口定位                                        │
│     - 跟随父窗口移动/关闭                                     │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 🔧 基本用法

### 1. 创建基础窗口

```javascript
// main.js
const { app, BrowserWindow } = require('electron')
const path = require('path')

let mainWindow

function createWindow() {
  mainWindow = new BrowserWindow({
    width: 1000,
    height: 700,
    webPreferences: {
      preload: path.join(__dirname, 'preload.js'),
      contextIsolation: true
    }
  })

  mainWindow.loadFile('index.html')

  // 窗口关闭时清理引用
  mainWindow.on('closed', () => {
    mainWindow = null
  })
}

app.whenReady().then(createWindow)
```

### 3. 多窗口管理

```javascript
// window-manager.js
class WindowManager {
  constructor() {
    this.windows = new Map()
  }

  // 创建窗口
  create(name, options) {
    const win = new BrowserWindow(options)

    this.windows.set(name, win)

    win.on('closed', () => {
      this.windows.delete(name)
    })

    return win
  }

  // 获取窗口
  get(name) {
    return this.windows.get(name)
  }

  // 获取所有窗口
  getAll() {
    return Array.from(this.windows.values())
  }

  // 关闭指定窗口
  close(name) {
    const win = this.windows.get(name)
    if (win) win.close()
  }

  // 关闭所有窗口
  closeAll() {
    this.windows.forEach(win => win.close())
  }

  // 窗口间通信
  send(from, to, channel, data) {
    const targetWindow = this.windows.get(to)
    if (targetWindow) {
      targetWindow.webContents.send(channel, { from, data })
    }
  }
}

module.exports = new WindowManager()
```

**使用窗口管理器**

```javascript
// main.js
const windowManager = require('./window-manager')

// 创建主窗口
const mainWindow = windowManager.create('main', {
  width: 1000,
  height: 700,
  webPreferences: { preload: path.join(__dirname, 'preload.js') }
})
mainWindow.loadFile('index.html')

// 创建设置窗口
const settingsWindow = windowManager.create('settings', {
  width: 500,
  height: 400,
  parent: mainWindow,  // 设置父窗口
  modal: false,
  webPreferences: { preload: path.join(__dirname, 'preload.js') }
})
settingsWindow.loadFile('settings.html')

// 窗口间通信
windowManager.send('main', 'settings', 'config-updated', { theme: 'dark' })
```

### 4. 窗口状态持久化

```javascript
const { app, BrowserWindow } = require('electron')
const Store = require('electron-store')

const store = new Store()

function createWindow() {
  // 从存储中恢复窗口状态
  const windowState = store.get('windowState', {
    width: 1000,
    height: 700,
    x: undefined,
    y: undefined,
    isMaximized: false
  })

  const win = new BrowserWindow({
    width: windowState.width,
    height: windowState.height,
    x: windowState.x,
    y: windowState.y,
    webPreferences: {
      preload: path.join(__dirname, 'preload.js')
    }
  })

  // 如果之前是最大化状态，恢复最大化
  if (windowState.isMaximized) {
    win.maximize()
  }

  // 监听窗口状态变化
  const saveState = () => {
    const isMaximized = win.isMaximized()
    const bounds = win.getBounds()

    store.set('windowState', {
      width: bounds.width,
      height: bounds.height,
      x: bounds.x,
      y: bounds.y,
      isMaximized
    })
  }

  win.on('close', saveState)
  win.on('resize', saveState)
  win.on('move', saveState)
  win.on('maximize', saveState)
  win.on('unmaximize', saveState)

  return win
}
```

## 🔧 高级特性

### 1. 父子窗口

```javascript
const parent = new BrowserWindow({
  width: 800,
  height: 600
})

const child = new BrowserWindow({
  width: 400,
  height: 300,
  parent: parent,  // 设置父窗口
  modal: true      // 模态窗口
})

// 子窗口会始终在父窗口上方
// 最小化父窗口时，子窗口也会最小化
```

### 2. 窗口快照

```javascript
// 捕获窗口内容
async function captureWindow(win) {
  const image = await win.webContents.capturePage()
  const buffer = image.toPNG()
  return buffer
}

// 截取指定区域
async function captureRegion(win, rect) {
  const image = await win.webContents.capturePage(rect)
  return image.toDataURL()
}
```

### 3. 窗口聚焦与置顶

```javascript
// 设置窗口置顶
win.setAlwaysOnTop(true, 'floating')

// 取消置顶
win.setAlwaysOnTop(false)

// 聚焦窗口
win.focus()

// 显示但不聚焦
win.showInactive()

// 防止窗口失焦
win.setFocusable(false)
```

### 4. 全屏与 Kiosk 模式

```javascript
// 全屏模式
win.setFullScreen(true)

// Kiosk 模式 (完全全屏，无法退出)
win.setKiosk(true)

// 退出 Kiosk
win.setKiosk(false)
```

## 📊 窗口事件

| 事件 | 说明 | 触发时机 |
|------|------|----------|
| `close` | 窗口即将关闭 | 可以阻止关闭 |
| `closed` | 窗口已关闭 | 无法阻止 |
| `ready-to-show` | 准备好显示 | 内容加载完成 |
| `focus` | 获得焦点 | 点击窗口 |
| `blur` | 失去焦点 | 点击其他窗口 |
| `maximize` | 最大化 | 点击最大化按钮 |
| `unmaximize` | 取消最大化 | 从最大化恢复 |
| `minimize` | 最小化 | 点击最小化按钮 |
| `restore` | 恢复 | 从最小化恢复 |
| `resize` | 调整大小 | 拖拽边框 |
| `move` | 移动 | 拖拽窗口 |
