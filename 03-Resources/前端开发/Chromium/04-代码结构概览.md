## 🔨 构建系统

### GN (Generate Ninja)

GN 是 Chromium 的元构建系统，生成 Ninja 构建文件。

```python
# BUILD.gn 文件示例

# 可执行文件
executable("chrome") {
  sources = [
    "main.cc",
  ]

  deps = [
    ":chrome_resources",
    "//base",
    "//content",
  ]
}

# 静态库
static_library("browser") {
  sources = [
    "browser.cc",
    "browser.h",
  ]

  deps = [
    "//base",
    "//net",
  ]

  defines = [ "ENABLE_FEATURES" ]
}

# 组件 (共享库)
component("content") {
  sources = [
    "content_browser.cc",
  ]

  public_deps = [
    "//base",
  ]
}

# 模板
template("my_template") {
  executable(target_name) {
    sources = invoker.sources
    deps = invoker.deps
  }
}

# 使用模板
my_template("my_app") {
  sources = [ "app.cc" ]
  deps = [ "//base" ]
}
```

### 常用 GN 命令

```bash
# 生成构建文件
gn gen out/Debug

# 生成 Release 构建
gn gen out/Release --args="is_debug=false"

# 查看参数
gn args out/Debug --list

# 设置参数
gn args out/Debug
# 在编辑器中添加:
# is_debug = true
# is_component_build = true
# enable_nacl = false

# 查看目标依赖
gn deps out/Debug //chrome

# 查看目标描述
gn desc out/Debug //chrome

# 格式化 BUILD.gn
gn format BUILD.gn
```

### Ninja 编译

```bash
# 编译 Chrome
ninja -C out/Debug chrome

# 编译所有
ninja -C out/Debug all

# 并行编译 (使用所有 CPU)
ninja -C out/Debug -j$(nproc) chrome

# 编译特定目标
ninja -C out/Debug unit_tests

# 查看编译命令
ninja -C out/Debug -t commands chrome

# 清理
ninja -C out/Debug -t clean
```

### 构建配置

```python
# 常用构建参数 (args.gn)

# 调试构建
is_debug = true

# 组件构建 (加快链接速度)
is_component_build = true

# 符号级别
symbol_level = 2  # 0: 无, 1: 最小, 2: 完整

# 优化级别
optimize_for_size = false

# 目标平台
target_os = "linux"  # linux, mac, win, android
target_cpu = "x64"   # x86, x64, arm, arm64

# 功能开关
enable_nacl = false
enable_remoting = false
use_aura = true

# macOS 特定
mac_deployment_target = "10.13"
```

## 📏 代码风格

### C++ 风格指南

```cpp
// 命名约定

// 类名: PascalCase
class BrowserContext {};

// 函数名: PascalCase
void InitializeBrowser();

// 变量名: snake_case
int page_count;
std::string url_string;

// 常量: k + PascalCase
const int kMaxCacheSize = 1000;

// 成员变量: 下划线后缀
class Foo {
  int count_;           // 私有成员
  int public_count_;    // 公开成员 (不推荐)
};

// 枚举: PascalCase, 值用 k + PascalCase
enum class State {
  kInitialized,
  kLoading,
  kComplete,
};

// 使用 using 而非 typedef
using CallbackList = std::vector<base::OnceClosure>;

// 使用 auto 推导类型
auto result = ComputeResult();  // OK
auto* ptr = GetPointer();       // OK，指针显式标注
```

### 智能指针

```cpp
// 使用 std::unique_ptr 管理独占所有权
std::unique_ptr<Browser> browser = std::make_unique<Browser>();

// 使用 scoped_refptr 管理共享所有权 (Chromium 特有)
scoped_refptr<RefCountedData> data = base::MakeRefCounted<RefCountedData>();

// 使用 base::WeakPtr 实现弱引用
class Foo : public base::SupportsWeakPtr<Foo> {
  base::WeakPtr<Foo> GetWeakPtr() { return weak_factory_.GetWeakPtr(); }

  base::WeakPtrFactory<Foo> weak_factory_{this};
};
```

### 回调与绑定

```cpp
// 使用 base::BindOnce 和 base::BindRepeating

// 一次性回调
void OnResponse(base::OnceCallback<void(bool)> callback) {
  std::move(callback).Run(true);
}

// 可重复回调
void OnEvent(base::RepeatingCallback<void(int)> callback) {
  callback.Run(1);
  callback.Run(2);
}

// 绑定成员函数
class Handler {
  void Handle(int value) {}
};

Handler handler;
auto callback = base::BindOnce(&Handler::Handle,
                               base::Unretained(&handler));

// 绑定带弱指针
auto weak_callback = base::BindOnce(&Handler::Handle,
                                    weak_ptr_factory.GetWeakPtr());
```
