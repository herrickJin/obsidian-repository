### 使用示例

```javascript
// 使用 CDP 进行自动化

// 1. 启动 Chrome 并开启远程调试
// chrome --remote-debugging-port=9222

// 2. 连接到 Chrome
const WebSocket = require('ws')
const ws = new WebSocket('ws://localhost:9222/devtools/page/xxx')

let id = 1

// 发送命令
function send(method, params = {}) {
  ws.send(JSON.stringify({ id: id++, method, params }))
}

// 处理响应
ws.on('message', (data) => {
  const message = JSON.parse(data)
  console.log('Received:', message)
})

ws.on('open', () => {
  // 导航到页面
  send('Page.navigate', { url: 'https://example.com' })

  // 启用网络监控
  send('Network.enable')

  // 启用页面事件
  send('Page.enable')

  // 执行 JavaScript
  send('Runtime.evaluate', {
    expression: 'document.title'
  })

  // 截图
  send('Page.captureScreenshot')
})

// 监听事件
ws.on('message', (data) => {
  const message = JSON.parse(data)
  if (message.method === 'Network.requestWillBeSent') {
    console.log('Request:', message.params.request.url)
  }
  if (message.method === 'Network.responseReceived') {
    console.log('Response:', message.params.response.status)
  }
})
```

### Puppeteer 使用 CDP

```javascript
// Puppeteer 封装了 CDP

const puppeteer = require('puppeteer')

async function main() {
  const browser = await puppeteer.launch()
  const page = await browser.newPage()

  // 启用请求拦截
  await page.setRequestInterception(true)
  page.on('request', (request) => {
    console.log('Request:', request.url())
    request.continue()
  })

  // 监听响应
  page.on('response', (response) => {
    console.log('Response:', response.status(), response.url())
  })

  // 访问页面
  await page.goto('https://example.com')

  // 执行 JavaScript
  const title = await page.evaluate(() => document.title)
  console.log('Title:', title)

  // 截图
  await page.screenshot({ path: 'screenshot.png' })

  // 使用 CDP 直接调用
  const client = await page.target().createCDPSession()
  await client.send('Network.enable')

  // 性能追踪
  await client.send('Tracing.start', {
    categories: 'devtools.timeline'
  })
  // 执行操作...
  await client.send('Tracing.end')

  await browser.close()
}
```

### 命令行追踪

```bash
# 启动时追踪
chrome --trace-startup=* --trace-startup-duration=5s \
       --trace-startup-file=trace.json

# 追踪到文件
chrome --trace-to-file=trace.json

# 指定类别
chrome --enable-tracing=browser,renderer,gpu

# Chrome for Testing
# 使用更简单的命令行追踪
```

### 分析追踪结果

```javascript
// 编程方式分析追踪数据

const fs = require('fs')
const trace = JSON.parse(fs.readFileSync('trace.json', 'utf8'))

// 提取特定事件
const events = trace.traceEvents.filter(e =>
  e.name === 'Paint' || e.name === 'Layout'
)

// 计算耗时
events.forEach(event => {
  if (event.ph === 'X') {  // 完整事件
    console.log(`${event.name}: ${event.dur / 1000}ms`)
  }
})

// 分析 V8 编译
const v8Compile = trace.traceEvents.filter(e =>
  e.name === 'V8.Compile' || e.name === 'V8.CompileCode'
)

// 分析布局
const layouts = trace.traceEvents.filter(e =>
  e.name === 'Layout'
)

// 分析绘制
const paints = trace.traceEvents.filter(e =>
  e.name === 'Paint'
)
```

### Breakpad 堆栈分析

```bash
# 使用 breakpad 分析 minidump

# 1. 生成符号文件
dump_syms out/Release/chrome > chrome.sym

# 2. 组织符号目录结构
# symbols/chrome/<version>/<os>/<cpu>/chrome.sym

# 3. 分析 minidump
minidump_stackwalk crash.dmp symbols/

# 输出示例:
# Operating system: Linux
# CPU: amd64
# Crash reason: SIGSEGV
# Crash address: 0x0000000000000000
#
# Thread 0 (crashed)
#  0  chrome!content::RenderViewImpl::OnMessageReceived [render_view_impl.cc:1234]
#  1  chrome!content::RenderThreadImpl::OnMessageReceived [render_thread_impl.cc:567]
```
