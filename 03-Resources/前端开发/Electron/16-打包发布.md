## 🔧 配置详解

### 1. 基本配置

```javascript
// electron-builder.yml (或使用 yml 文件)

appId: com.example.myapp
productName: My App
copyright: Copyright © 2024 My Company

directories:
  output: dist/release
  buildResources: build

# 包含/排除文件
files:
  - "**/*"
  - "!**/*.{ts,tsx,map}"
  - "!src/**/*"
  - "!test/**/*"
  - "!**/node_modules/.cache"
  - from: "dist"
    to: "app"
    filter:
      - "**/*"

# ASAR 打包
asar: true
asarUnpack:
  - "**/*.node"  # 原生模块不打包
```

### 2. Windows 配置

```javascript
win:
  target:
    - target: nsis
      arch:
        - x64
        - ia32
    - target: portable
      arch:
        - x64
  icon: build/icon.ico
  artifactName: ${productName}-${version}-${arch}.${ext}

# NSIS 安装程序
nsis:
  oneClick: false
  allowToChangeInstallationDirectory: true
  allowElevation: true
  installerIcon: build/icon.ico
  uninstallerIcon: build/icon.ico
  installerHeaderIcon: build/icon.ico
  createDesktopShortcut: true
  createStartMenuShortcut: true
  shortcutName: ${productName}
  include: build/installer.nsh  # 自定义脚本
  deleteAppDataOnUninstall: false
  runAfterFinish: true

# 代码签名 (Windows)
certificateFile: path/to/certificate.pfx
certificatePassword: ${env.CERT_PASSWORD}
signingHashAlgorithms:
  - sha256
  - sha1
```

### 3. macOS 配置

```javascript
mac:
```

## 🔄 CI/CD 集成

### 1. GitHub Actions

```yaml
# .github/workflows/build.yml
name: Build and Release

on:
  push:
    tags:
      - 'v*'

jobs:
  build:
    runs-on: ${{ matrix.os }}

    strategy:
      matrix:
        os: [macos-latest, windows-latest, ubuntu-latest]

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Build application
        run: npm run build

      - name: Build Electron app
        run: npm run electron:build
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      - name: Upload artifacts
        uses: actions/upload-artifact@v4
        with:
          name: ${{ matrix.os }}-build
          path: dist/

  release:
    needs: build
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Download all artifacts
        uses: actions/download-artifact@v4
        with:
          path: dist/

      - name: Create Release
        uses: softprops/action-gh-release@v1
        with:
          files: dist/**/*
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```

### 2. 自动发布

```yaml
# 完整的发布流程
name: Release

on:
  push:
    tags:
      - 'v*'

jobs:
  release:
    runs-on: ${{ matrix.os }}

    strategy:
      matrix:
        os: [macos-latest, windows-latest, ubuntu-latest]

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Install dependencies
        run: npm ci

      - name: Build
        run: npm run build

      # macOS 签名和公证
      - name: Prepare macOS signing
        if: matrix.os == 'macos-latest'
        run: |
          echo "Setting up macOS signing..."
        env:
          CSC_LINK: ${{ secrets.MAC_CERT }}
          CSC_KEY_PASSWORD: ${{ secrets.MAC_CERT_PASSWORD }}

      - name: Build Electron
        run: npm run electron:build -- --publish always
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          CSC_LINK: ${{ secrets.MAC_CERT }}
          CSC_KEY_PASSWORD: ${{ secrets.MAC_CERT_PASSWORD }}
          # Windows 签名
          WIN_CSC_LINK: ${{ secrets.WIN_CERT }}
          WIN_CSC_KEY_PASSWORD: ${{ secrets.WIN_CERT_PASSWORD }}
```

## 🍎 macOS 公证

### 1. 准备工作

```bash
# 1. 获取开发者证书
# 2. 创建 App-specific password
# 3. 获取 Team ID
```

### 2. 公证脚本

```javascript
// build/notarize.js
const { notarize } = require('@electron/notarize')

exports.default = async function notarizing(context) {
  const { electronPlatformName, appOutDir } = context

  if (electronPlatformName !== 'darwin') {
    return
  }

  const appName = context.packager.appInfo.productFilename

  return await notarize({
    tool: 'notarytool',
    appBundleId: 'com.example.myapp',
    appPath: `${appOutDir}/${appName}.app`,
    appleId: process.env.APPLE_ID,
    appleIdPassword: process.env.APPLE_ID_PASSWORD,
    teamId: process.env.APPLE_TEAM_ID
  })
}
```
