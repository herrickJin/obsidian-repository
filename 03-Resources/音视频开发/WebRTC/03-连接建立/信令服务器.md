# 信令服务器

> 信令服务器是 WebRTC 中负责交换 SDP 和 ICE 候选的服务端，是建立连接的桥梁

## 📌 核心作用

- 交换 SDP Offer/Answer
- 传输 ICE 候选
- 管理房间和用户
- 处理用户加入/离开

---

## 🔧 WebSocket 信令服务器

### Node.js + ws 实现

```javascript
// server.js
const WebSocket = require('ws');
const wss = new WebSocket.Server({ port: 8080 });

// 房间管理
const rooms = new Map();

wss.on('connection', (ws) => {
  console.log('客户端连接');

  ws.on('message', (message) => {
    const data = JSON.parse(message);
    handleMessage(ws, data);
  });

  ws.on('close', () => {
    handleLeave(ws);
  });
});

function handleMessage(ws, data) {
  switch (data.type) {
    case 'join':
      handleJoin(ws, data);
      break;
    case 'offer':
    case 'answer':
    case 'candidate':
      forwardMessage(ws, data);
      break;
  }
}

function handleJoin(ws, data) {
  const { roomId, userId } = data;

  // 保存用户信息
  ws.roomId = roomId;
  ws.userId = userId;

  // 加入房间
  if (!rooms.has(roomId)) {
    rooms.set(roomId, new Set());
  }
  rooms.get(roomId).add(ws);

  // 通知房间内其他用户
  broadcastToRoom(ws, {
    type: 'user-joined',
    userId: userId
  }, true);

  // 告知当前用户房间内已有用户
  const existingUsers = [];
  rooms.get(roomId).forEach(client => {
    if (client !== ws) {
      existingUsers.push(client.userId);
    }
  });

  ws.send(JSON.stringify({
    type: 'room-users',
    users: existingUsers
  }));
}

function forwardMessage(ws, data) {
  const { targetId } = data;
  const room = rooms.get(ws.roomId);

  if (room) {
    room.forEach(client => {
      if (client.userId === targetId) {
        client.send(JSON.stringify({
          ...data,
          from: ws.userId
        }));
      }
    });
  }
}

function broadcastToRoom(ws, data, excludeSelf = false) {
  const room = rooms.get(ws.roomId);

  if (room) {
    const message = JSON.stringify(data);
    room.forEach(client => {
      if (!excludeSelf || client !== ws) {
        client.send(message);
      }
    });
  }
}

function handleLeave(ws) {
  const room = rooms.get(ws.roomId);

  if (room) {
    room.delete(ws);

    // 通知其他用户
    broadcastToRoom(ws, {
      type: 'user-left',
      userId: ws.userId
    }, true);

    // 如果房间为空，删除房间
    if (room.size === 0) {
      rooms.delete(ws.roomId);
    }
  }
}

console.log('信令服务器运行在 ws://localhost:8080');
```

### 客户端连接

```javascript
// client.js
class SignalingClient {
  constructor(serverUrl) {
    this.ws = null;
    this.callbacks = {};
  }

  connect() {
    return new Promise((resolve, reject) => {
      this.ws = new WebSocket(serverUrl);

      this.ws.onopen = () => {
        console.log('连接信令服务器成功');
        resolve();
      };

      this.ws.onerror = (error) => {
        reject(error);
      };

      this.ws.onmessage = (event) => {
        const data = JSON.parse(event.data);
        this.handleMessage(data);
      };

      this.ws.onclose = () => {
        console.log('连接关闭');
        if (this.callbacks.onClose) {
          this.callbacks.onClose();
        }
      };
    });
  }

  handleMessage(data) {
    switch (data.type) {
      case 'user-joined':
        if (this.callbacks.onUserJoined) {
          this.callbacks.onUserJoined(data);
        }
        break;

      case 'user-left':
        if (this.callbacks.onUserLeft) {
          this.callbacks.onUserLeft(data);
        }
        break;

      case 'room-users':
        if (this.callbacks.onRoomUsers) {
          this.callbacks.onRoomUsers(data);
        }
        break;

      case 'offer':
        if (this.callbacks.onOffer) {
          this.callbacks.onOffer(data);
        }
        break;

      case 'answer':
        if (this.callbacks.onAnswer) {
          this.callbacks.onAnswer(data);
        }
        break;

      case 'candidate':
        if (this.callbacks.onCandidate) {
          this.callbacks.onCandidate(data);
        }
        break;
    }
  }

  // 加入房间
  join(roomId, userId) {
    this.send({
      type: 'join',
      roomId,
      userId
    });
  }

  // 发送 Offer
  sendOffer(targetId, offer) {
    this.send({
      type: 'offer',
      targetId,
      sdp: offer
    });
  }

  // 发送 Answer
  sendAnswer(targetId, answer) {
    this.send({
      type: 'answer',
      targetId,
      sdp: answer
    });
  }

  // 发送 ICE 候选
  sendCandidate(targetId, candidate) {
    this.send({
      type: 'candidate',
      targetId,
      candidate
    });
  }

  send(data) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(data));
    }
  }

  on(event, callback) {
    this.callbacks[event] = callback;
  }

  close() {
    if (this.ws) {
      this.ws.close();
    }
  }
}
```

## 🔌 Socket.IO 实现

### 服务端

```javascript
// server.js
const io = require('socket.io')(8080, {
  cors: {
    origin: '*'
  }
});

io.on('connection', (socket) => {
  console.log('用户连接:', socket.id);

  socket.on('join', (data) => {
    const { roomId, userId } = data;

    socket.join(roomId);
    socket.roomId = roomId;
    socket.userId = userId;

    // 通知房间内其他人
    socket.to(roomId).emit('user-joined', {
      userId,
      socketId: socket.id
    });

    // 获取房间内用户列表
    const clients = io.sockets.adapter.rooms.get(roomId);
    const users = [];
    if (clients) {
      clients.forEach(clientId => {
        const client = io.sockets.sockets.get(clientId);
        if (client && client.userId) {
          users.push({
            userId: client.userId,
            socketId: client.id
          });
        }
      });
    }

    socket.emit('room-users', { users });
  });

  socket.on('offer', (data) => {
    socket.to(data.targetId).emit('offer', {
      from: socket.id,
      sdp: data.sdp
    });
  });

  socket.on('answer', (data) => {
    socket.to(data.targetId).emit('answer', {
      from: socket.id,
      sdp: data.sdp
    });
  });

  socket.on('candidate', (data) => {
    socket.to(data.targetId).emit('candidate', {
      from: socket.id,
      candidate: data.candidate
    });
  });

  socket.on('disconnect', () => {
    if (socket.roomId) {
      socket.to(socket.roomId).emit('user-left', {
        userId: socket.userId,
        socketId: socket.id
      });
    }
  });
});

console.log('Socket.IO 服务器运行在 http://localhost:8080');
```

import { io } from 'socket.io-client';

class SignalingClient {
  constructor(serverUrl) {
    this.socket = io(serverUrl);
    this.setupListeners();
  }

  setupListeners() {
    this.socket.on('connect', () => {
      console.log('连接成功:', this.socket.id);
    });

    this.socket.on('user-joined', (data) => {
      if (this.onUserJoined) this.onUserJoined(data);
    });

    this.socket.on('user-left', (data) => {
      if (this.onUserLeft) this.onUserLeft(data);
    });

    this.socket.on('room-users', (data) => {
      if (this.onRoomUsers) this.onRoomUsers(data);
    });

    this.socket.on('offer', (data) => {
      if (this.onOffer) this.onOffer(data);
    });

    this.socket.on('answer', (data) => {
      if (this.onAnswer) this.onAnswer(data);
    });

    this.socket.on('candidate', (data) => {
      if (this.onCandidate) this.onCandidate(data);
    });
  }

  join(roomId, userId) {
    this.socket.emit('join', { roomId, userId });
  }

  sendOffer(targetId, sdp) {
    this.socket.emit('offer', { targetId, sdp });
  }

  sendAnswer(targetId, sdp) {
    this.socket.emit('answer', { targetId, sdp });
  }

  sendCandidate(targetId, candidate) {
    this.socket.emit('candidate', { targetId, candidate });
  }

  disconnect() {
    this.socket.disconnect();
  }
}

## 🚀 完整 WebRTC 客户端

```javascript
class WebRTCClient {
  constructor(signalingServer) {
    this.signaling = signalingServer;
    this.peers = new Map(); // userId -> RTCPeerConnection
    this.localStream = null;
    this.config = {
      iceServers: [
        { urls: 'stun:stun.l.google.com:19302' }
      ]
    };

    this.setupSignalingCallbacks();
  }

  setupSignalingCallbacks() {
    this.signaling.onUserJoined = async (data) => {
      console.log('用户加入:', data.userId);
      // 新用户加入时，主动发起连接
      await this.createPeerConnection(data.userId, true);
    };

    this.signaling.onUserLeft = (data) => {
      console.log('用户离开:', data.userId);
      this.closePeerConnection(data.userId);
    };

    this.signaling.onRoomUsers = async (data) => {
      console.log('房间用户:', data.users);
      // 加入房间时，向已有用户发起连接
      for (const user of data.users) {
        await this.createPeerConnection(user.userId, true);
      }
    };

    this.signaling.onOffer = async (data) => {
      console.log('收到 Offer:', data.from);
      await this.handleOffer(data);
    };

    this.signaling.onAnswer = async (data) => {
      console.log('收到 Answer:', data.from);
      await this.handleAnswer(data);
    };

    this.signaling.onCandidate = async (data) => {
      console.log('收到 Candidate:', data.from);
      await this.handleCandidate(data);
    };
  }

  async setLocalStream(stream) {
    this.localStream = stream;
  }

  async createPeerConnection(userId, isInitiator) {
    if (this.peers.has(userId)) return;

    const pc = new RTCPeerConnection(this.config);

    // 添加本地流
    if (this.localStream) {
      this.localStream.getTracks().forEach(track => {
        pc.addTrack(track, this.localStream);
      });
    }

    // ICE 候选
    pc.onicecandidate = (event) => {
      if (event.candidate) {
        this.signaling.sendCandidate(userId, event.candidate);
      }
    };

    // 远程流
    pc.ontrack = (event) => {
      console.log('收到远程流:', userId);
      if (this.onRemoteStream) {
        this.onRemoteStream(userId, event.streams[0]);
      }
    };

    // 连接状态
    pc.onconnectionstatechange = () => {
      console.log(`连接状态 ${userId}:`, pc.connectionState);
      if (this.onConnectionStateChange) {
        this.onConnectionStateChange(userId, pc.connectionState);
      }
    };

    this.peers.set(userId, pc);

    // 如果是发起者，创建 Offer
    if (isInitiator) {
      const offer = await pc.createOffer();
      await pc.setLocalDescription(offer);
      this.signaling.sendOffer(userId, offer);
    }
  }

  async handleOffer(data) {
    const { from, sdp } = data;

    // 确保有 PeerConnection
    if (!this.peers.has(from)) {
      await this.createPeerConnection(from, false);
    }

    const pc = this.peers.get(from);

    await pc.setRemoteDescription(new RTCSessionDescription(sdp));

    const answer = await pc.createAnswer();
    await pc.setLocalDescription(answer);

    this.signaling.sendAnswer(from, answer);
  }

  async handleAnswer(data) {
    const { from, sdp } = data;
    const pc = this.peers.get(from);

    if (pc) {
      await pc.setRemoteDescription(new RTCSessionDescription(sdp));
    }
  }

  async handleCandidate(data) {
    const { from, candidate } = data;
    const pc = this.peers.get(from);

    if (pc) {
      await pc.addIceCandidate(new RTCIceCandidate(candidate));
    }
  }

  closePeerConnection(userId) {
    const pc = this.peers.get(userId);
    if (pc) {
      pc.close();
      this.peers.delete(userId);
    }
  }

  closeAll() {
    this.peers.forEach((pc) => pc.close());
    this.peers.clear();
    this.signaling.disconnect();
  }
}

// 使用示例
async function main() {
  // 获取本地流
  const localStream = await navigator.mediaDevices.getUserMedia({
    video: true,
    audio: true
  });

  // 连接信令服务器
  const signaling = new SignalingClient('ws://localhost:8080');
  await signaling.connect();

  // 创建 WebRTC 客户端
  const webrtc = new WebRTCClient(signaling);
  await webrtc.setLocalStream(localStream);

  // 设置回调
  webrtc.onRemoteStream = (userId, stream) => {
    const video = document.createElement('video');
    video.srcObject = stream;
    video.autoplay = true;
    video.playsInline = true;
    document.body.appendChild(video);
  };

  // 加入房间
  signaling.join('room-123', 'user-' + Date.now());
}
```
