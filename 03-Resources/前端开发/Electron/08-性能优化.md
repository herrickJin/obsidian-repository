```javascript
  performanceMetrics[name] = value
  console.log(`Metric ${name}:`, value - performanceMetrics.appStartTime)
})
```

## 🚀 启动速度优化

### 1. 延迟加载

```javascript
// ❌ 不好的做法：一开始就加载所有模块
const heavyModule = require('./heavy-module')
const anotherModule = require('./another-module')

// ✅ 好的做法：按需加载
let heavyModule = null

function getHeavyModule() {
  if (!heavyModule) {
    heavyModule = require('./heavy-module')
  }
  return heavyModule
}
```

### 2. 延迟创建窗口

```javascript
// ✅ 只在需要时创建窗口
let settingsWindow = null

ipcMain.handle('openSettings', () => {
  if (settingsWindow) {
    settingsWindow.focus()
    return
  }

  settingsWindow = new BrowserWindow({
    width: 500,
    height: 400
  })

  settingsWindow.loadFile('settings.html')

  settingsWindow.on('closed', () => {
    settingsWindow = null
  })
})
```

### 3. 预加载优化

```javascript
// 使用 show: false 延迟显示窗口
function createWindow() {
  const win = new BrowserWindow({
    show: false,  // 先不显示
    webPreferences: {
      preload: path.join(__dirname, 'preload.js')
    }
  })

  win.loadFile('index.html')

  // 窗口准备好后再显示
  win.once('ready-to-show', () => {
    win.show()
  })
}
```

### 4. V8 代码缓存

```javascript
// 使用 v8-compile-cache
// 安装: npm install v8-compile-cache

// 在主进程入口最顶部

```

### 5. 减少启动时 I/O

```javascript
// ❌ 同步读取文件
const config = JSON.parse(fs.readFileSync('config.json', 'utf-8'))

// ✅ 异步读取或使用缓存
let configCache = null

async function getConfig() {
  if (configCache) return configCache

  const content = await fs.readFile('config.json', 'utf-8')
  configCache = JSON.parse(content)
  return configCache
}
```

## 💾 内存优化

### 1. 进程模型优化

```javascript
// 减少渲染进程数量
const win1 = new BrowserWindow({
  webPreferences: {
    // 同一域名共享进程
    webPreferences: {
      additionalArguments: ['--shared-process']
    }
  }
})
```

### 2. 及时释放资源

```javascript
// 关闭窗口时清理资源
let heavyResource = null

mainWindow.on('closed', () => {
  // 清理资源
  if (heavyResource) {
    heavyResource.destroy()
    heavyResource = null
  }

  // 取消事件监听
  ipcMain.removeAllListeners('some-event')

  mainWindow = null
})
```

### 3. 避免内存泄漏

```javascript
// ❌ 内存泄漏：忘记移除监听器
ipcMain.on('event', (event, data) => {
  const handler = () => console.log(data)
  mainWindow.on('resize', handler)
})

// ✅ 正确做法：保存引用并移除
const handlers = new Map()

ipcMain.on('event', (event, data) => {
  const handler = () => console.log(data)
  handlers.set(event.sender.id, handler)
  mainWindow.on('resize', handler)
})

mainWindow.on('closed', () => {
  handlers.forEach((handler, id) => {
    mainWindow.removeListener('resize', handler)
  })
  handlers.clear()
})
```

### 4. 限制缓存大小

```javascript
// 限制 webContents 缓存
mainWindow.webContents.session.clearCache()

// 或设置缓存大小
app.commandLine.appendSwitch('disk-cache-size', '104857600')  // 100MB
```

```javascript
// 使用 Intersection Observer
const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      const img = entry.target
      img.src = img.dataset.src
      observer.unobserve(img)

  })
})
```

### 3. 防抖和节流

```javascript
// 防抖 - 搜索输入
function debounce(func, wait) {
  let timeout
  return function executedFunction(...args) {
    const later = () => {
      clearTimeout(timeout)
      func(...args)
    }
    clearTimeout(timeout)
    timeout = setTimeout(later, wait)
  }
}

// 节流 - 滚动事件
function throttle(func, limit) {
  let inThrottle
  return function(...args) {
    if (!inThrottle) {
      func.apply(this, args)
      inThrottle = true
      setTimeout(() => inThrottle = false, limit)
    }
  }
}

// 使用
searchInput.addEventListener('input', debounce(search, 300))
window.addEventListener('scroll', throttle(handleScroll, 100))
```
