# 码率自适应

> 码率自适应 (ABR) 是根据网络状况动态调整视频/音频码率的技术

## 📌 核心目标

- 最大化利用带宽
- 保证视频流畅
- 避免卡顿和冻结
- 平衡质量和流畅性

## 📊 码率分配策略

### 带宽分配

```javascript
class BandwidthAllocator {
  constructor() {
    // 分配比例
    this.ratios = {
      audio: 0.1,      // 音频 10%
      video: 0.7,      // 视频 70%
      fec: 0.1,        // FEC 10%
      retransmit: 0.1  // 重传 10%
    };
  }

  allocate(totalBandwidth) {
    // 优先保证音频
    const audioBitrate = Math.min(
      totalBandwidth * this.ratios.audio,
      64000  // 最大 64kbps
    );

    // 剩余带宽
    const remaining = totalBandwidth - audioBitrate;

    // FEC 和重传
    const fecBitrate = remaining * (this.ratios.fec / (this.ratios.fec + this.ratios.retransmit + this.ratios.video));
    const retransmitBitrate = remaining * (this.ratios.retransmit / (this.ratios.fec + this.ratios.retransmit + this.ratios.video));

    // 视频使用剩余
    const videoBitrate = remaining - fecBitrate - retransmitBitrate;

    return {
      audio: audioBitrate,
      video: videoBitrate,
      fec: fecBitrate,
      retransmit: retransmitBitrate,
      total: totalBandwidth
    };
  }
}
```

### 动态调整

```javascript
class DynamicBitrateController {
  constructor() {
    this.currentBitrate = 500000;
    this.targetBitrate = 500000;
    this.minBitrate = 30000;
    this.maxBitrate = 5000000;

    // 调整参数
    this.increaseFactor = 1.05;
    this.decreaseFactor = 0.7;
    this.smoothFactor = 0.3;
  }

  update(networkEstimate) {
    // 根据网络估计调整目标码率
    if (networkEstimate < this.targetBitrate) {
      // 网络带宽不足，快速降低
      this.targetBitrate = Math.max(
        networkEstimate * 0.9,
        this.minBitrate
      );
    } else {
      // 网络充足，缓慢增加
      this.targetBitrate = Math.min(
        this.targetBitrate * this.increaseFactor,
        this.maxBitrate,
        networkEstimate * 0.95
      );
    }

    // 平滑过渡到目标码率
    this.currentBitrate += (this.targetBitrate - this.currentBitrate) * this.smoothFactor;

    return this.currentBitrate;
  }
}
```

## 🎬 视频码率控制

### 编码器码率控制

```javascript
// WebRTC 使用 VP8/VP9/H.264 的码率控制
// 可以通过 setParameters 调整

async function setVideoBitrate(sender, bitrate) {
  const parameters = sender.getParameters();

  if (!parameters.encodings) {
    parameters.encodings = [{}];
  }

  // 设置码率范围
  parameters.encodings[0].maxBitrate = bitrate;
  parameters.encodings[0].minBitrate = Math.floor(bitrate * 0.5);

  await sender.setParameters(parameters);
}
```

```javascript
async function setupSimulcast(pc, track) {
  const transceiver = pc.addTransceiver(track, {

    sendEncodings: [
      {
        rid: 'f',  // Full resolution

      {
```

## 🔊 音频码率控制

### Opus 码率范围

```javascript
// Opus 支持的码率范围: 6kbps - 510kbps
// 常用配置:

const opusBitrates = {
  narrowBand: 8000,      // 窄带语音
  wideBand: 16000,       // 宽带语音
  superWideBand: 24000,  // 超宽带
  fullBand: 48000,       // 全带
  highQuality: 64000,    // 高质量音乐
  stereo: 96000          // 立体声
};

// 设置 Opus 码率
async function setOpusBitrate(pc, bitrate) {
  const transceivers = pc.getTransceivers();
  const audioTransceiver = transceivers.find(t => t.receiver.track.kind === 'audio');

  if (audioTransceiver) {
    const sender = audioTransceiver.sender;
    const parameters = sender.getParameters();

    // 设置最大码率
    parameters.encodings[0].maxBitrate = bitrate;

    await sender.setParameters(parameters);
  }
}
```

### 音频优先级

```javascript
// 在带宽不足时优先保证音频
class AudioPriorityController {
  constructor() {
    this.audioMinBitrate = 16000;
    this.audioMaxBitrate = 32000;
  }

  adjust(totalBandwidth) {
    // 确保音频至少有最小带宽
    if (totalBandwidth < this.audioMinBitrate + 50000) {
      // 极低带宽情况
      return {
        audio: this.audioMinBitrate,
        video: Math.max(totalBandwidth - this.audioMinBitrate, 0)
      };
    }

    // 正常分配
    return {
      audio: Math.min(totalBandwidth * 0.1, this.audioMaxBitrate),
      video: totalBandwidth * 0.9
    };
  }
}
```

## 📈 码率阶梯

### 视频码率推荐值

```javascript
const videoBitrateLadder = [
  // { resolution, framerate, minBitrate, targetBitrate, maxBitrate }
  { width: 320, height: 180, fps: 15, min: 50000, target: 100000, max: 150000 },
  { width: 320, height: 240, fps: 15, min: 80000, target: 150000, max: 200000 },
  { width: 480, height: 270, fps: 24, min: 150000, target: 250000, max: 350000 },
  { width: 640, height: 360, fps: 24, min: 250000, target: 400000, max: 600000 },
  { width: 640, height: 480, fps: 30, min: 400000, target: 600000, max: 800000 },
  { width: 1280, height: 720, fps: 30, min: 800000, target: 1500000, max: 2500000 },
  { width: 1920, height: 1080, fps: 30, min: 1500000, target: 2500000, max: 4000000 },
  { width: 1920, height: 1080, fps: 60, min: 2500000, target: 4000000, max: 6000000 }
];

function selectQuality(bitrate) {
  // 选择不超过当前码率的最高质量
  for (let i = videoBitrateLadder.length - 1; i >= 0; i--) {
    if (bitrate >= videoBitrateLadder[i].min) {
      return videoBitrateLadder[i];
    }
  }
  return videoBitrateLadder[0];
}
```

## 🎛️ 完整实现

```javascript
class AdaptiveBitrateManager {
  constructor(pc) {
    this.pc = pc;
    this.allocator = new BandwidthAllocator();
    this.currentQuality = null;
    this.lastAdjustTime = 0;
    this.adjustInterval = 3000; // 最小调整间隔
  }

  async adjust(networkStats) {
    const now = Date.now();

    // 限制调整频率
    if (now - this.lastAdjustTime < this.adjustInterval) {
      return;
    }

    // 获取带宽估计
    const bandwidth = networkStats.bandwidth || 500000;

    // 分配带宽
    const allocation = this.allocator.allocate(bandwidth);

    // 选择视频质量
    const videoQuality = selectQuality(allocation.video);

    // 应用码率
    await this.applyBitrate(allocation, videoQuality);

    this.lastAdjustTime = now;
    this.currentQuality = videoQuality;
  }

  async applyBitrate(allocation, quality) {
    const senders = this.pc.getSenders();

    for (const sender of senders) {
      const track = sender.track;
      if (!track) continue;

      const parameters = sender.getParameters();
      if (!parameters.encodings) {
        parameters.encodings = [{}];
      }

      if (track.kind === 'video') {
        // 视频码率
        parameters.encodings[0].maxBitrate = quality.max;
        parameters.encodings[0].minBitrate = quality.min;

        // 分辨率和帧率
        await track.applyConstraints({
          width: { ideal: quality.width },
          height: { ideal: quality.height },
          frameRate: { ideal: quality.fps }
        });
      } else if (track.kind === 'audio') {
        // 音频码率
        parameters.encodings[0].maxBitrate = allocation.audio;
      }

      await sender.setParameters(parameters);
    }
  }

  getCurrentQuality() {
    return this.currentQuality;
  }
}
```
