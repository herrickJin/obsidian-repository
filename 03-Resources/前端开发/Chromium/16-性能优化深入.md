### 内存优化技术

```javascript
// JavaScript 内存优化

// 1. 对象池
class ObjectPool {
  constructor(factory, reset) {
    this.factory = factory
    this.reset = reset
    this.pool = []
  }

  acquire() {
    return this.pool.length > 0
      ? this.pool.pop()
      : this.factory()
  }

  release(obj) {
    this.reset(obj)
    this.pool.push(obj)
  }
}

// 2. 避免闭包内存泄漏
class EventHandler {
  constructor() {
    // ✅ 使用箭头函数绑定
    this.onClick = (e) => this.handleClick(e)
  }

  // 而不是
  // constructor() {
  //   this.onClick = this.handleClick.bind(this)  // 每次创建新函数
  // }
}

// 3. 大数据处理
// ❌ 一次性加载所有数据
const allData = loadData()  // 可能很大

// ✅ 分批处理
async function* processData(stream) {
  for await (const chunk of stream) {
    yield processChunk(chunk)
  }
}

// 4. 使用 TypedArray
// ❌
const arr = new Array(1000000)
// ✅
const arr = new Float64Array(1000000)

// 5. 及时释放
function processLargeFile() {
  const data = loadLargeFile()
  const result = process(data)
  data = null  // 释放引用
  return result
}
```

### 优化技术

```javascript
// CPU 性能优化技术

// 1. 代码分割和懒加载
// ❌
import { heavyFunction } from './heavy-module'

// ✅
const { heavyFunction } = await import('./heavy-module')

// 2. Web Worker
const worker = new Worker('compute.js')
worker.postMessage(largeData)
worker.onmessage = (e) => {
  // 处理结果
}

// 3. requestIdleCallback
function processLowPriority(tasks) {
  requestIdleCallback((deadline) => {
    while (deadline.timeRemaining() > 0 && tasks.length > 0) {
      processTask(tasks.shift())
    }
    if (tasks.length > 0) {
      requestIdleCallback(processLowPriority)
    }
  })
}

// 4. 防抖和节流
function debounce(fn, delay) {
  let timer
  return (...args) => {
    clearTimeout(timer)
    timer = setTimeout(() => fn(...args), delay)
  }
}

function throttle(fn, limit) {
  let inThrottle
  return (...args) => {
    if (!inThrottle) {
      fn(...args)
      inThrottle = true
      setTimeout(() => inThrottle = false, limit)
    }
  }
}

// 5. 虚拟列表
class VirtualList {
  constructor(container, itemHeight, renderItem) {
    this.container = container
    this.itemHeight = itemHeight
    this.renderItem = renderItem
    this.visibleStart = 0
    this.visibleEnd = 0
  }

  render(items) {
    const scrollTop = this.container.scrollTop
    const viewportHeight = this.container.clientHeight

    this.visibleStart = Math.floor(scrollTop / this.itemHeight)
    this.visibleEnd = Math.min(
      items.length,
      Math.ceil((scrollTop + viewportHeight) / this.itemHeight)
    )

    // 只渲染可见项
    const fragment = document.createDocumentFragment()
    for (let i = this.visibleStart; i < this.visibleEnd; i++) {
      fragment.appendChild(this.renderItem(items[i], i))
    }
    this.container.innerHTML = ''
    this.container.appendChild(fragment)
  }
}

// 6. 批量 DOM 操作
// ❌
for (let i = 0; i < 1000; i++) {
  const div = document.createElement('div')
  container.appendChild(div)
}

// ✅
const fragment = document.createDocumentFragment()
for (let i = 0; i < 1000; i++) {
  const div = document.createElement('div')
  fragment.appendChild(div)
}
container.appendChild(fragment)
```

// JavaScript 渲染优化

// 1. 使用 requestAnimationFrame
function animate() {
  // 读取布局信息
  const width = element.offsetWidth

  requestAnimationFrame(() => {
    // 写入样式
    element.style.transform = `translateX(${width}px)`
  })
}

// 2. 避免布局抖动
// ❌ 读写交替
for (let i = 0; i < elements.length; i++) {
  elements[i].style.width = elements[i].offsetWidth + 10 + 'px'
}

// ✅ 分离读写
const widths = elements.map(el => el.offsetWidth)
elements.forEach((el, i) => {
  el.style.width = widths[i] + 10 + 'px'
})

// 3. 使用 ResizeObserver
const observer = new ResizeObserver((entries) => {
  for (const entry of entries) {
    // 响应大小变化
  }
})
observer.observe(element)

// 4. Intersection Observer
const intersectionObserver = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      // 元素进入视口

  })

intersectionObserver.observe(lazyElement)

### 性能 API

```javascript
// Performance API 使用

// 1. Navigation Timing
const timing = performance.timing
const metrics = {
  dns: timing.domainLookupEnd - timing.domainLookupStart,
  tcp: timing.connectEnd - timing.connectStart,
  ttfb: timing.responseStart - timing.requestStart,
  download: timing.responseEnd - timing.responseStart,
  domReady: timing.domContentLoadedEventEnd - timing.navigationStart,
  load: timing.loadEventEnd - timing.navigationStart
}

// 2. Resource Timing
const resources = performance.getEntriesByType('resource')
resources.forEach(resource => {
  console.log({
    name: resource.name,
    duration: resource.duration,
    transferSize: resource.transferSize
  })
})

// 3. User Timing
performance.mark('start')
// 执行操作
performance.mark('end')
performance.measure('operation', 'start', 'end')
const measure = performance.getEntriesByName('operation')[0]
console.log(measure.duration)

// 4. Long Tasks
const observer = new PerformanceObserver((list) => {
  list.getEntries().forEach(entry => {
    console.log('Long Task:', entry.duration, 'ms')
  })
})
observer.observe({ entryTypes: ['longtask'] })

// 5. Layout Instability (CLS)
let clsValue = 0
const layoutShiftObserver = new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    if (!entry.hadRecentInput) {
      clsValue += entry.value
    }
  }
})
layoutShiftObserver.observe({ type: 'layout-shift', buffered: true })
```
