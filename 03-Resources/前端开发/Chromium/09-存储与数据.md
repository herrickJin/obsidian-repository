### Chromium SQLite 封装

```cpp
// Chromium 中的 SQLite 封装

// sql::Database
class Database {
 public:
  // 打开数据库
  bool Open(const base::FilePath& path);

  // 执行 SQL
  bool Execute(const char* sql);
  bool ExecuteWithTimeout(const char* sql, base::TimeDelta timeout);

  // 准备语句
  std::unique_ptr<Statement> GetUniqueStatement(const char* sql);

  // 事务
  bool BeginTransaction();
  bool CommitTransaction();
  void RollbackTransaction();
};

// sql::Statement
class Statement {
 public:
  // 绑定参数
  void BindString(int param, const std::string& value);
  void BindInt(int param, int value);
  void BindInt64(int param, int64_t value);
  void BindDouble(int param, double value);
  void BindBlob(int param, const void* value, int size);

  // 执行
  bool Step();  // 移动到下一行
  bool Run();   // 执行不返回结果

  // 获取结果
  std::string ColumnString(int col);
  int ColumnInt(int col);
  int64_t ColumnInt64(int col);
  double ColumnDouble(int col);
};

// 使用示例
sql::Database db;
db.Open(db_path);

sql::Statement stmt(db.GetUniqueStatement(
    "SELECT * FROM cookies WHERE host_key = ?"));
stmt.BindString(0, "example.com");

while (stmt.Step()) {
  std::string name = stmt.ColumnString(1);
  std::string value = stmt.ColumnString(2);
  // 处理...
}
```

### Chromium LevelDB 封装

```cpp
// leveldb::DB 接口

// 打开数据库
leveldb::DB* db;
leveldb::Options options;
options.create_if_missing = true;
leveldb::Status status = leveldb::DB::Open(options, "/path/to/db", &db);

// 写入
leveldb::WriteOptions write_options;
write_options.sync = true;  // 同步写入
db->Put(write_options, "key", "value");

// 读取
leveldb::ReadOptions read_options;
std::string value;
leveldb::Status status = db->Get(read_options, "key", &value);

// 删除
db->Delete(write_options, "key");

// 批量写入
leveldb::WriteBatch batch;
batch.Put("key1", "value1");
batch.Put("key2", "value2");
batch.Delete("key3");
db->Write(write_options, &batch);

// 遍历
leveldb::Iterator* it = db->NewIterator(read_options);
for (it->SeekToFirst(); it->Valid(); it->Next()) {
  std::string key = it->key().ToString();
  std::string value = it->value().ToString();
}
delete it;

// 关闭
delete db;
```

### IndexedDB 使用示例

```javascript
// 打开数据库
const request = indexedDB.open('MyDatabase', 1)

request.onupgradeneeded = (event) => {
  const db = event.target.result

  // 创建对象存储
  if (!db.objectStoreNames.contains('users')) {
    const store = db.createObjectStore('users', { keyPath: 'id' })

    // 创建索引
    store.createIndex('email', 'email', { unique: true })
    store.createIndex('name', 'name', { unique: false })
  }
}

request.onsuccess = (event) => {
  const db = event.target.result

  // 添加数据
  const transaction = db.transaction(['users'], 'readwrite')
  const store = transaction.objectStore('users')

  store.add({ id: 1, name: 'Alice', email: 'alice@example.com' })
  store.add({ id: 2, name: 'Bob', email: 'bob@example.com' })

  // 查询数据
  const getRequest = store.get(1)
  getRequest.onsuccess = (e) => {
    console.log(e.target.result)
  }

  // 使用索引查询
  const index = store.index('email')
  const emailRequest = index.get('alice@example.com')
  emailRequest.onsuccess = (e) => {
    console.log(e.target.result)
  }

  // 使用游标遍历
  const cursorRequest = store.openCursor()
  cursorRequest.onsuccess = (e) => {
    const cursor = e.target.result
    if (cursor) {
      console.log(cursor.value)
      cursor.continue()
    }
  }
}
```

## 🍪 Cookie 存储

### Cookie 数据库结构

```sql
-- Cookies 表结构 (简化)
CREATE TABLE cookies (
  creation_utc INTEGER NOT NULL,
  host_key TEXT NOT NULL,      -- 域名
  name TEXT NOT NULL,          -- Cookie 名称
  value TEXT NOT NULL,         -- Cookie 值
  path TEXT NOT NULL,          -- 路径
  expires_utc INTEGER NOT NULL,-- 过期时间
  is_secure INTEGER NOT NULL,  -- Secure 标志
  is_httponly INTEGER NOT NULL,-- HttpOnly 标志
  last_access_utc INTEGER NOT NULL,
  has_expires INTEGER NOT NULL,
  is_persistent INTEGER NOT NULL,
  priority INTEGER NOT NULL,
  encrypted_value BLOB,        -- 加密值 (可选)
  samesite INTEGER NOT NULL,   -- SameSite 属性
  source_scheme INTEGER NOT NULL,
  source_port INTEGER NOT NULL,
  is_same_party INTEGER NOT NULL,
  last_update_utc INTEGER NOT NULL
);

CREATE UNIQUE INDEX cookies_unique_index
ON cookies(host_key, name, path);
```
