## 📖 原生模块基础

### 什么是原生模块？

```
┌─────────────────────────────────────────────────────────────┐
│                    原生模块架构                              │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  JavaScript (Node.js)                                       │
│         ↓                                                   │
│  N-API / Node-Addon-API                                     │
│         ↓                                                   │
│  C/C++ 原生代码                                              │
│         ↓                                                   │
│  操作系统 API / 第三方库                                     │
│                                                             │
└─────────────────────────────────────────────────────────────┘

优势:
- 性能: C++ 比 JS 快 10-100 倍
- 能力: 访问系统底层 API
- 复用: 使用现有的 C/C++ 库
```

### 4. C++ 实现

```cpp
// src/addon.cc
#include <napi.h>

// 简单函数
Napi::String Hello(const Napi::CallbackInfo& info) {
  Napi::Env env = info.Env();
  return Napi::String::New(env, "Hello from C++!");
}

// 带参数的函数
Napi::Number Add(const Napi::CallbackInfo& info) {
  Napi::Env env = info.Env();

  if (info.Length() < 2 || !info[0].IsNumber() || !info[1].IsNumber()) {
    Napi::TypeError::New(env, "Two numbers expected").ThrowAsJavaScriptException();
    return Napi::Number::New(env, 0);
  }

  double arg1 = info[0].As<Napi::Number>().DoubleValue();
  double arg2 = info[1].As<Napi::Number>().DoubleValue();

  return Napi::Number::New(env, arg1 + arg2);
}

// 异步函数
class AsyncWorker : public Napi::AsyncWorker {
public:
  AsyncWorker(Napi::Function& callback, int input)
    : Napi::AsyncWorker(callback), input(input), result(0) {}

  void Execute() override {
    // 在工作线程执行
    result = input * input;  // 模拟耗时计算
  }

  void OnOK() override {
    Napi::HandleScope scope(Env());
    Callback().Call({Env().Null(), Napi::Number::New(Env(), result)});
  }

private:
  int input;
  int result;
};

Napi::Value AsyncSquare(const Napi::CallbackInfo& info) {
  Napi::Env env = info.Env();

  if (info.Length() < 2 || !info[0].IsNumber() || !info[1].IsFunction()) {
    Napi::TypeError::New(env, "Number and callback expected").ThrowAsJavaScriptException();
    return env.Null();
  }

  int input = info[0].As<Napi::Number>().Int32Value();
  Napi::Function callback = info[1].As<Napi::Function>();

  AsyncWorker* worker = new AsyncWorker(callback, input);
  worker->Queue();

  return env.Null();
}

// 类
class Calculator : public Napi::ObjectWrap<Calculator> {
public:
  static Napi::Object Init(Napi::Env env, Napi::Object exports) {
    Napi::HandleScope scope(env);

    Napi::Function func = DefineClass(env, "Calculator", {
      InstanceMethod("add", &Calculator::Add),
      InstanceMethod("subtract", &Calculator::Subtract),
      InstanceAccessor("value", &Calculator::GetValue, &Calculator::SetValue)
    });

    Napi::FunctionReference* constructor = new Napi::FunctionReference();
    *constructor = Napi::Persistent(func);
    env.SetInstanceData(constructor);

    exports.Set("Calculator", func);
    return exports;
  }

  Calculator(const Napi::CallbackInfo& info) : Napi::ObjectWrap<Calculator>(info) {
    Napi::Env env = info.Env();
    value_ = info.Length() > 0 ? info[0].As<Napi::Number>().DoubleValue() : 0;
  }

private:
  double value_;

  Napi::Value Add(const Napi::CallbackInfo& info) {
    value_ += info[0].As<Napi::Number>().DoubleValue();
    return Napi::Number::New(info.Env(), value_);
  }

  Napi::Value Subtract(const Napi::CallbackInfo& info) {
    value_ -= info[0].As<Napi::Number>().DoubleValue();
    return Napi::Number::New(info.Env(), value_);
  }

  Napi::Value GetValue(const Napi::CallbackInfo& info) {
    return Napi::Number::New(info.Env(), value_);
  }

  void SetValue(const Napi::CallbackInfo& info, const Napi::Value& value) {
    value_ = value.As<Napi::Number>().DoubleValue();
  }
};

// 模块初始化
Napi::Object Init(Napi::Env env, Napi::Object exports) {
  exports.Set(Napi::String::New(env, "hello"), Napi::Function::New(env, Hello));
  exports.Set(Napi::String::New(env, "add"), Napi::Function::New(env, Add));
  exports.Set(Napi::String::New(env, "asyncSquare"), Napi::Function::New(env, AsyncSquare));

  Calculator::Init(env, exports);

  return exports;
}

NODE_API_MODULE(addon, Init)
```

### 3. Rust 实现

```rust
// native/src/lib.rs
use neon::prelude::*;

// 简单函数
fn hello(mut cx: FunctionContext) -> JsResult<JsString> {
    Ok(cx.string("Hello from Rust!"))
}

// 带参数的函数
fn add(mut cx: FunctionContext) -> JsResult<JsNumber> {
    let a = cx.argument::<JsNumber>(0)?.value(&mut cx);
    let b = cx.argument::<JsNumber>(1)?.value(&mut cx);
    Ok(cx.number(a + b))
}

// 异步函数
fn async_compute(mut cx: FunctionContext) -> JsResult<JsUndefined> {
    let n = cx.argument::<JsNumber>(0)?.value(&mut cx) as i64;
    let callback = cx.argument::<JsFunction>(1)?.root(&mut cx);

    // 使用 neon 的异步任务
    let channel = cx.channel();
    std::thread::spawn(move || {
        let result = n * n;  // 耗时计算

        channel.send(move |mut cx| {
            let callback = callback.into_inner(&mut cx);
            let this = cx.undefined();
            let args: Vec<Handle<JsValue>> = vec![
                cx.null().upcast(),
                cx.number(result).upcast(),
            ];
            callback.call(&mut cx, this, args)?;
            Ok(())
        });
    });

    Ok(cx.undefined())
}

// 类
struct Calculator {
    value: f64,
}

impl Finalize for Calculator {}

impl Calculator {
    fn new(mut cx: FunctionContext) -> JsResult<JsBox<Calculator>> {
        let value = if cx.len() > 0 {
            cx.argument::<JsNumber>(0)?.value(&mut cx)
        } else {
            0.0
        };
        Ok(cx.boxed(Calculator { value }))
    }

    fn add(mut cx: FunctionContext) -> JsResult<JsNumber> {
        let this = cx.this().downcast_or_throw::<JsBox<Calculator>, _>(&mut cx)?;
        let n = cx.argument::<JsNumber>(0)?.value(&mut cx);

        let mut calculator = this.borrow_mut();
        calculator.value += n;

        Ok(cx.number(calculator.value))
    }

    fn get_value(mut cx: FunctionContext) -> JsResult<JsNumber> {
        let this = cx.this().downcast_or_throw::<JsBox<Calculator>, _>(&mut cx)?;
        let calculator = this.borrow();
        Ok(cx.number(calculator.value))
    }
}

// 注册模块
#[neon::main]
fn main(mut cx: ModuleContext) -> NeonResult<()> {
    cx.export_function("hello", hello)?;
    cx.export_function("add", add)?;
    cx.export_function("asyncCompute", async_compute)?;
    cx.export_class::<Calculator>("Calculator")?;
    Ok(())
}
```

## 📦 Electron 中使用原生模块

### 1. 重新编译

```bash
# 为 Electron 重新编译原生模块
npm install --save-dev @electron/rebuild

# 重新编译
npx @electron/rebuild

# 或在 package.json 中配置
{
  "scripts": {
    "postinstall": "electron-rebuild"
  }
}
```

### 2. electron-builder 配置

```json
{
  "build": {
    "npmRebuild": true,
    "asarUnpack": [
      "**/*.node"
    ]
  }
}
```

### 3. 使用原生模块

```javascript
// main.js
const myAddon = require('my-native-addon')

// 同步调用
const result = myAddon.add(1, 2)

// 异步调用
myAddon.asyncCompute(100, (err, result) => {
  console.log(result)
})
```
