## 🖥️ 平台检测

### 1. 基本检测

```javascript
const os = require('os')

// 操作系统
const platform = process.platform
// 'darwin' (macOS)
// 'win32' (Windows)
// 'linux' (Linux)

// 架构
const arch = process.arch
// 'x64' | 'arm64' | 'ia32'

// 便捷判断
const isMac = process.platform === 'darwin'
const isWindows = process.platform === 'win32'
const isLinux = process.platform === 'linux'
const isArm64 = process.arch === 'arm64'
```

### 2. 平台工具函数

```javascript
// utils/platform.js
module.exports = {
  isMac: process.platform === 'darwin',
  isWindows: process.platform === 'win32',
  isLinux: process.platform === 'linux',
  isArm64: process.arch === 'arm64',
  isX64: process.arch === 'x64',

  // 获取平台名称
  getPlatformName() {
    switch (process.platform) {
      case 'darwin': return 'macOS'
      case 'win32': return 'Windows'
      case 'linux': return 'Linux'
      default: return 'Unknown'
    }
  },

  // 获取快捷键修饰符
  getModifierKey() {
    return this.isMac ? 'Command' : 'Control'
  },

  // 获取应用图标路径
  getAppIcon() {
    if (this.isWindows) return 'build/icon.ico'
    if (this.isMac) return 'build/icon.icns'
    return 'build/icons/256x256.png'
  }
}
```

### 2. Dock

```javascript
const { app } = require('electron')

if (process.platform === 'darwin') {
  // 设置 Dock 图标
  app.dock.setIcon('/path/to/icon.png')

  // 设置 Dock 菜单
  const dockMenu = Menu.buildFromTemplate([
    { label: '新建窗口', click: () => createWindow() },
    { label: '新建文件', click: () => createFile() }
  ])
  app.dock.setMenu(dockMenu)

  // 设置 Dock 徽章
  app.dock.setBadge('3')  // 显示数字徽章

  // 隐藏 Dock 图标
  app.dock.hide()

  // 显示 Dock 图标
  app.dock.show()

  // 弹跳 Dock 图标
  app.dock.bounce()  // 一次性弹跳
  const id = app.dock.bounce('critical')  // 持续弹跳
  app.dock.cancelBounce(id)  // 取消弹跳
}
```

### 4. Touch Bar

```javascript
const { TouchBar } = require('electron')

const { TouchBarButton, TouchBarLabel, TouchBarSpacer } = TouchBar

const button = new TouchBarButton({
  label: '🎉 拍照',
  backgroundColor: '#7851A9',
  click: () => {
    console.log('Button clicked')
  }

  label: 'Hello Touch Bar'

const touchBar = new TouchBar({
  items: [
    button,
    new TouchBarSpacer({ size: 'large' }),
    label

if (process.platform === 'darwin') {
  win.setTouchBar(touchBar)
}
```

### 5. 通知

## 🪟 Windows 特有功能

### 1. 任务栏

```javascript
const { app, BrowserWindow } = require('electron')

if (process.platform === 'win32') {
  const win = BrowserWindow.getAllWindows()[0]

  // 设置任务栏图标
  win.setOverlayIcon('/path/to/icon.png', '描述')

  // 任务栏进度条
  win.setProgressBar(0.5)  // 0-1
  win.setProgressBar(-1)   // 不确定进度
  win.setProgressBar(-2)   // 移除进度条

  // 任务栏按钮 (Thumb Buttons)
  win.setThumbarButtons([
    {
      tooltip: '播放',
      icon: '/path/to/play.png',
      click: () => console.log('播放')
    },
    {
      tooltip: '暂停',
      icon: '/path/to/pause.png',
      click: () => console.log('暂停')
    }
  ])

  // 跳转列表
  app.setUserTasks([
    {
      program: process.execPath,
      arguments: '--new-window',
      iconPath: process.execPath,
      iconIndex: 0,
      title: '新建窗口',
      description: '打开新窗口'
    }
  ])
}
```

### 2. 窗口效果

## 🐧 Linux 特有功能

### 1. 桌面环境检测

```javascript
const fs = require('fs')

// 检测桌面环境
function getDesktopEnvironment() {
  const xdgCurrentDesktop = process.env.XDG_CURRENT_DESKTOP
  const gnomeSession = process.env.GNOME_DESKTOP_SESSION_ID
  const kdeSession = process.env.KDE_FULL_SESSION

  if (xdgCurrentDesktop) {
    if (xdgCurrentDesktop.includes('GNOME')) return 'GNOME'
    if (xdgCurrentDesktop.includes('KDE')) return 'KDE'
    if (xdgCurrentDesktop.includes('XFCE')) return 'XFCE'
  }

  if (gnomeSession) return 'GNOME'
  if (kdeSession) return 'KDE'

  return 'Unknown'
}
```

## 🎨 跨平台样式

### 1. CSS 平台适配

```css
/* 使用 CSS 变量 */
:root {
  --font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif;
```

### 3. 快捷键适配

```javascript
// 使用 CmdOrCtrl
const { Menu } = require('electron')

const template = [
  {
    label: '复制',
    accelerator: 'CmdOrCtrl+C',  // macOS: Cmd+C, Windows: Ctrl+C
    role: 'copy'
  },
  {
    label: '粘贴',
    accelerator: 'CmdOrCtrl+V',
    role: 'paste'
  }
]

// 显示快捷键
function formatAccelerator(accelerator) {
  if (process.platform === 'darwin') {
    return accelerator
      .replace('CmdOrCtrl', '⌘')
      .replace('CommandOrControl', '⌘')
      .replace('Shift', '⇧')
      .replace('Alt', '⌥')
      .replace('Control', '⌃')
  }
  return accelerator
    .replace('CmdOrCtrl', 'Ctrl')
    .replace('CommandOrControl', 'Ctrl')
}
```
