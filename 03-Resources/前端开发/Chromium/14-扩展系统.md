### Manifest V3

```json
// manifest.json (Manifest V3)

{
  "manifest_version": 3,
  "name": "My Extension",
  "version": "1.0",
  "description": "A sample extension",

  // 权限
  "permissions": [
    "storage",
    "activeTab",
    "scripting"
  ],

  // 可选权限
  "optional_permissions": [
    "bookmarks",
    "history"
  ],

  // 主机权限
  "host_permissions": [
    "https://*.example.com/*"
  ],

  // Background Service Worker
  "background": {
    "service_worker": "background.js"
  },

  // Content Scripts
  "content_scripts": [
    {
      "matches": ["https://*.example.com/*"],
      "js": ["content.js"],
      "css": ["style.css"]
    }
  ],

  // Popup
  "action": {
    "default_popup": "popup.html",
    "default_icon": {
      "16": "icons/16.png",
      "48": "icons/48.png",
      "128": "icons/128.png"
    }
  },

  // Options Page
  "options_page": "options.html",

  // Web Accessible Resources
  "web_accessible_resources": [
    {
      "resources": ["images/*"],
      "matches": ["<all_urls>"]
    }
  ]
}
```

## 📜 Background Script

### Service Worker (Manifest V3)

```javascript
// background.js - Service Worker

// 安装事件
chrome.runtime.onInstalled.addListener((details) => {
  if (details.reason === 'install') {
    console.log('Extension installed')
    // 初始化设置
    chrome.storage.local.set({ enabled: true })
  } else if (details.reason === 'update') {
    console.log('Extension updated')
  }
})

// 启动事件
chrome.runtime.onStartup.addListener(() => {
  console.log('Browser started')
})

// 消息处理
chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
  console.log('Received message:', message)

  if (message.type === 'getData') {
    fetchData(message.url).then(data => {
      sendResponse({ success: true, data })
    })
    return true  // 保持消息通道开放
  }
})

// 标签页事件
chrome.tabs.onCreated.addListener((tab) => {
  console.log('Tab created:', tab.id)
})

chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (changeInfo.status === 'complete') {
    console.log('Tab loaded:', tab.url)
  }
})

// 网络请求
chrome.webRequest.onBeforeRequest.addListener(
  (details) => {
    console.log('Request:', details.url)
    // 可以取消或重定向请求
  },
  { urls: ['<all_urls>'] }
)

// 上下文菜单
chrome.contextMenus.create({
  id: 'my-menu',
  title: 'My Menu Item',
  contexts: ['selection']
})

chrome.contextMenus.onClicked.addListener((info, tab) => {
  console.log('Menu clicked:', info.selectionText)
})
```

## 📄 Content Script

### 注入和执行

```javascript
// content.js - Content Script

// Content Script 运行在网页上下文中
// 可以访问 DOM，但与页面 JavaScript 隔离

// DOM 操作
const button = document.createElement('button')
button.textContent = 'Click Me'
button.addEventListener('click', () => {
  console.log('Button clicked')
  // 发送消息到 Background
  chrome.runtime.sendMessage({ type: 'buttonClicked' })
})
document.body.appendChild(button)

// 监听来自 Background 的消息
chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
  if (message.type === 'highlight') {
    highlightText(message.text)
    sendResponse({ success: true })
  }
})

// 与页面脚本通信 (通过 CustomEvent)
window.addEventListener('message-from-extension', (event) => {
  console.log('Message from page:', event.detail)
})

// 向页面发送消息
window.dispatchEvent(new CustomEvent('message-to-page', {
  detail: { data: 'from extension' }
}))

// 高亮文本函数
function highlightText(text) {
  const walker = document.createTreeWalker(
    document.body,
    NodeFilter.SHOW_TEXT,
    null
  )

  const nodes = []
  while (walker.nextNode()) {
    if (walker.currentNode.textContent.includes(text)) {
      nodes.push(walker.currentNode)
    }
  }

  nodes.forEach(node => {
    const span = document.createElement('span')
    span.style.backgroundColor = 'yellow'
    span.textContent = node.textContent
    node.parentNode.replaceChild(span, node)
  })
}
```

### 注入方式

```javascript
// 方式 1: Manifest 声明
// manifest.json
{
  "content_scripts": [{
    "matches": ["https://*.example.com/*"],
    "js": ["content.js"],
    "run_at": "document_idle"  // document_start, document_end, document_idle
  }]
}

// 方式 2: 动态注入 (Background Script)
chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (changeInfo.status === 'complete' && tab.url.includes('example.com')) {
    chrome.scripting.executeScript({
      target: { tabId },
      files: ['content.js']
    })
  }
})

// 方式 3: 通过 Action 点击注入
chrome.action.onClicked.addListener((tab) => {
  chrome.scripting.executeScript({
    target: { tabId: tab.id },
    func: () => {
      document.body.style.border = '5px solid red'
    }
  })
})
```

## 🔌 扩展 API

### 常用 API

```javascript
// Storage API
chrome.storage.local.get(['key'], (result) => {
  console.log(result.key)
})

chrome.storage.local.set({ key: 'value' }, () => {
  console.log('Saved')
})

chrome.storage.onChanged.addListener((changes, namespace) => {
  console.log('Storage changed:', changes)
})

// Tabs API
chrome.tabs.query({ active: true, currentWindow: true }, (tabs) => {
  console.log('Active tab:', tabs[0].url)
})

chrome.tabs.create({ url: 'https://example.com' })

chrome.tabs.update(tabId, { url: 'https://example.com' })

// Runtime API (消息)
chrome.runtime.sendMessage({ type: 'ping' }, (response) => {
  console.log('Response:', response)
})

chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
  sendResponse({ received: true })
})

// Scripting API (Manifest V3)
chrome.scripting.executeScript({
  target: { tabId: tabId },
  files: ['script.js']
})

chrome.scripting.insertCSS({
  target: { tabId: tabId },
  files: ['style.css']
})

// Bookmarks API
chrome.bookmarks.create({
  parentId: '1',
  title: 'My Bookmark',
  url: 'https://example.com'
})

// Notifications API
chrome.notifications.create({
  type: 'basic',
  iconUrl: 'icon.png',
  title: 'Notification',
  message: 'Hello from extension!'
})
```

## 🔒 扩展安全

### 安全隔离

```
┌─────────────────────────────────────────────────────────────┐
│                   扩展安全模型                               │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  隔离环境:                                                   │
│  ├── Background Script: 独立扩展进程                        │
│  ├── Content Script: 隔离的 JavaScript 世界                 │
│  └── Page Script: 网页自己的 JavaScript                     │
│                                                             │
│  Content Script 隔离:                                        │
│  ├── 不能访问页面变量                                       │
│  ├── 不能调用页面函数                                       │
│  └── DOM 共享，但 JavaScript 隔离                           │
│                                                             │
│  CSP 限制:                                                   │
│  ├── 禁止内联脚本                                           │
│  ├── 禁止 eval                                              │
│  └── 只能加载扩展内资源                                     │
│                                                             │
│  远程代码:                                                   │
│  └── Manifest V3 禁止远程代码                               │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 安全最佳实践

```javascript
// ✅ 安全: 使用消息传递
chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
  // 验证发送者
  if (!sender.tab || !sender.tab.url.startsWith('https://example.com')) {
    return  // 忽略不信任的来源
  }

  // 验证消息内容
  if (typeof message.data !== 'string') {
    return
  }

  // 处理消息
  sendResponse({ result: processData(message.data) })
})

// ❌ 不安全: 执行不受信任的代码
chrome.tabs.executeScript({
  code: userInput  // 危险！
})

// ✅ 安全: 使用参数化方式
chrome.tabs.executeScript({
  file: 'script.js'
})

// ✅ 安全: 验证输入
function validateInput(input) {
  if (typeof input !== 'string') return null
  if (input.length > 1000) return null
  if (/<script>/i.test(input)) return null
  return input
}
```
