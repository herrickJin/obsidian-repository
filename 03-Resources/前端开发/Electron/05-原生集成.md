## 📁 文件系统操作

### 1. 基本文件操作

```javascript
// main.js
const fs = require('fs').promises
const path = require('path')

// 通过 IPC 暴露文件操作
ipcMain.handle('file:read', async (event, filePath) => {
  try {
    const content = await fs.readFile(filePath, 'utf-8')
    return { success: true, content }
  } catch (error) {
    return { success: false, error: error.message }
  }
})

ipcMain.handle('file:write', async (event, filePath, content) => {
  try {
    await fs.writeFile(filePath, content, 'utf-8')
    return { success: true }
  } catch (error) {
    return { success: false, error: error.message }
  }
})

ipcMain.handle('file:delete', async (event, filePath) => {
  try {
    await fs.unlink(filePath)
    return { success: true }
  } catch (error) {
    return { success: false, error: error.message }
  }
})

ipcMain.handle('file:exists', async (event, filePath) => {
  try {
    await fs.access(filePath)
    return true
  } catch {
    return false
  }
})
```

### 2. 文件监听

```javascript
const chokidar = require('chokidar')

// 监听文件变化
ipcMain.handle('file:watch', (event, filePath) => {
  const watcher = chokidar.watch(filePath, {
    ignored: /(^|[\/\\])\../,  // 忽略隐藏文件
    persistent: true
  })

  watcher
    .on('change', path => {
      event.sender.send('file:changed', path)
    })
    .on('add', path => {
      event.sender.send('file:added', path)
    })
    .on('unlink', path => {
      event.sender.send('file:deleted', path)
    })

  return true
})
```

## 📋 剪贴板 API

### 1. 基本剪贴板操作

```javascript
const { clipboard, nativeImage } = require('electron')

// 读取文本
ipcMain.handle('clipboard:readText', () => {
  return clipboard.readText()
})

// 写入文本
ipcMain.handle('clipboard:writeText', (event, text) => {
  clipboard.writeText(text)
  return true
})

// 读取 HTML
ipcMain.handle('clipboard:readHTML', () => {
  return clipboard.readHTML()
})

// 写入 HTML
ipcMain.handle('clipboard:writeHTML', (event, html) => {
  clipboard.writeHTML(html)
  return true
})

// 读取图片
ipcMain.handle('clipboard:readImage', () => {
  const image = clipboard.readImage()
  if (image.isEmpty()) return null
  return image.toDataURL()
})

// 写入图片
ipcMain.handle('clipboard:writeImage', (event, dataURL) => {
  const image = nativeImage.createFromDataURL(dataURL)
  clipboard.writeImage(image)
  return true
})

// 清空剪贴板
ipcMain.handle('clipboard:clear', () => {
  clipboard.clear()
  return true
})
```

### 2. 高级剪贴板操作

```javascript
// 写入多种格式
ipcMain.handle('clipboard:write', (event, { text, html, image }) => {
  clipboard.write({
    text: text,
    html: html,
    image: image ? nativeImage.createFromDataURL(image) : undefined
  })
  return true
})

// 可用格式
ipcMain.handle('clipboard:availableFormats', () => {
  return clipboard.availableFormats()
})

// 读取书签 (macOS)
ipcMain.handle('clipboard:readBookmark', () => {
  return clipboard.readBookmark()
  // 返回: { title: string, url: string }
})

// 写入书签 (macOS)
ipcMain.handle('clipboard:writeBookmark', (event, title, url) => {
  clipboard.writeBookmark(title, url)
  return true
})
```

## 🔔 桌面通知

### 1. 基本通知

```javascript
const { Notification } = require('electron')

// 检查通知权限
ipcMain.handle('notification:isSupported', () => {
  return Notification.isSupported()
})

// 显示通知
function showNotification(title, body, options = {}) {
  const notification = new Notification({
    title: title,
    body: body,
    icon: options.icon,           // 图标路径
    silent: options.silent,       // 是否静音
    urgency: options.urgency,     // Linux: 'normal' | 'critical' | 'low'
    closeButtonText: options.closeButtonText  // macOS: 关闭按钮文字
  })

  notification.on('click', () => {
    // 点击通知时的处理
    if (mainWindow) {
      mainWindow.show()
      mainWindow.focus()
    }
  })

  notification.on('close', () => {
    // 通知关闭时的处理
    console.log('Notification closed')
  })

  notification.show()
  return notification
}

ipcMain.handle('notification:show', (event, options) => {
  return showNotification(options.title, options.body, options)
})
```

### 2. 带操作的通知

```javascript
// 带回复输入的通知 (macOS)
ipcMain.handle('notification:showWithReply', (event, options) => {
  const notification = new Notification({
    title: options.title,
    body: options.body,
    hasReply: true,              // 显示回复输入框
    replyPlaceholder: '回复...',  // 占位文字
    actions: [
      { type: 'button', text: '接受' },
      { type: 'button', text: '拒绝' }
    ]
  })

  notification.on('reply', (event, reply) => {
    console.log('用户回复:', reply)
    // 处理用户回复
  })

  notification.on('action', (event, index) => {
    console.log('用户点击了操作:', index)
    // 处理按钮点击
  })

  notification.show()
})
```

### 3. 进度通知

```javascript
// 带进度条的通知
function showProgressNotification(title, body, progress) {
  const notification = new Notification({
    title: title,
    body: body,
    progress: progress  // 0-100
  })

  notification.show()
  return notification
}

// 更新通知进度
let downloadNotification

ipcMain.handle('notification:startDownload', (event, filename) => {
  downloadNotification = new Notification({
    title: '下载中',
    body: filename,
    progress: 0
  })
  downloadNotification.show()
})

ipcMain.handle('notification:updateProgress', (event, progress) => {
  if (downloadNotification) {
    // 重新创建通知以更新进度
    downloadNotification.close()
    downloadNotification = new Notification({
      title: '下载中',
      body: `已完成 ${progress}%`,
      progress: progress
    })
    downloadNotification.show()
  }
})
```

## 🔌 电源监控

### 1. 电源状态

```javascript
const { powerMonitor } = require('electron')

// 获取电源状态
ipcMain.handle('power:getState', () => {
  return {
    charging: powerMonitor.isOnBatteryPower() === false,
    batteryLevel: powerMonitor.getSystemIdleState(30),
    idleTime: powerMonitor.getSystemIdleTime()
  }
})

// 监听电源状态变化
powerMonitor.on('on-ac', () => {
  console.log('接通电源')
  mainWindow?.webContents.send('power:changed', { charging: true })
})

powerMonitor.on('on-battery', () => {
  console.log('使用电池')
  mainWindow?.webContents.send('power:changed', { charging: false })
})
```

### 2. 系统空闲检测

```javascript
// 获取系统空闲时间（秒）
ipcMain.handle('power:getIdleTime', () => {
  return powerMonitor.getSystemIdleTime()
})

// 获取空闲状态
ipcMonitor.on('power:getIdleState', (event, idleThreshold = 60) => {
  // 返回: 'active' | 'idle' | 'locked' | 'unknown'
  return powerMonitor.getSystemIdleState(idleThreshold)
})
```

### 3. 系统事件监听

```javascript
// 系统挂起/恢复
powerMonitor.on('suspend', () => {
  console.log('系统即将挂起')
  // 保存数据
  saveBeforeSuspend()
})

powerMonitor.on('resume', () => {
  console.log('系统恢复')
  // 恢复操作
  restoreAfterResume()
})

// 锁屏/解锁 (macOS/Windows)
powerMonitor.on('lock-screen', () => {
  console.log('屏幕已锁定')
})

powerMonitor.on('unlock-screen', () => {
  console.log('屏幕已解锁')
})

// 关机/重启 (Linux)
powerMonitor.on('shutdown', () => {
  console.log('系统即将关机')
  // 清理并允许关机
  app.quit()
})
```

## 📺 屏幕捕获

### 1. 获取屏幕信息

```javascript
const { desktopCapturer, screen } = require('electron')

// 获取所有显示器
ipcMain.handle('screen:getDisplays', () => {
  return screen.getAllDisplays().map(display => ({
    id: display.id,
    label: display.label,
    bounds: display.bounds,
    size: display.size,
    workArea: display.workArea,
    scaleFactor: display.scaleFactor,
    rotation: display.rotation,
    internal: display.internal
  }))
})

// 获取主显示器
ipcMain.handle('screen:getPrimaryDisplay', () => {
  return screen.getPrimaryDisplay()
})
```

### 2. 桌面捕获源

```javascript
// 获取可捕获的源（屏幕和窗口）
ipcMain.handle('desktopCapturer:getSources', async (event, options = {}) => {
  const sources = await desktopCapturer.getSources({
    types: options.types || ['screen', 'window'],
    thumbnailSize: options.thumbnailSize || { width: 150, height: 150 },
    fetchWindowIcons: options.fetchWindowIcons || false
  })

  return sources.map(source => ({
    id: source.id,           // 如 'screen:1' 或 'window:123'
    name: source.name,       // 显示名称
    thumbnail: source.thumbnail.toDataURL(),
    display_id: source.display_id,
    appIcon: source.appIcon?.toDataURL()
  }))
})
```

### 3. 在渲染进程中使用

```javascript
// preload.js
contextBridge.exposeInMainWorld('electronAPI', {
  getDesktopSources: (options) => ipcRenderer.invoke('desktopCapturer:getSources', options)
})

// renderer.js
async function startScreenShare() {
  const sources = await window.electronAPI.getDesktopSources({
    types: ['screen', 'window']
  })

  // 显示选择器让用户选择
  const selectedSource = await showSourceSelector(sources)

  // 获取媒体流
  const stream = await navigator.mediaDevices.getUserMedia({
    audio: false,
    video: {
      mandatory: {
        chromeMediaSource: 'desktop',
        chromeMediaSourceId: selectedSource.id,
        maxWidth: 1920,
        maxHeight: 1080
      }
    }
  })

  // 显示在 video 元素中
  const video = document.querySelector('video')
  video.srcObject = stream
  video.play()
}
```

### 4. 截图功能

```javascript
// 截取指定屏幕
ipcMain.handle('screen:capture', async (event, displayId) => {
  const sources = await desktopCapturer.getSources({
    types: ['screen'],
    thumbnailSize: screen.getPrimaryDisplay().size
  })

  const source = sources.find(s => s.display_id === displayId) || sources[0]
  if (source) {
    return source.thumbnail.toDataURL()
  }
  return null
})

// 截取窗口
ipcMain.handle('window:capture', async (event, windowId) => {
  const sources = await desktopCapturer.getSources({
    types: ['window'],
    thumbnailSize: { width: 1920, height: 1080 }
  })

  const source = sources.find(s => s.id === windowId)
  if (source) {
    return source.thumbnail.toDataURL()
  }
  return null
})
```
