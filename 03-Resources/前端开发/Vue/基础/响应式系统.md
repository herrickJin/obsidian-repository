# Vue 3 响应式系统

> 深入理解 Vue 3 响应式原理

## 📌 核心概念

Vue 3 使用 **Proxy** 替代了 Vue 2 的 `Object.defineProperty`，实现了完整的响应式系统。

### Vue 2 vs Vue 3 响应式对比

| 特性 | Vue 2 (defineProperty) | Vue 3 (Proxy) |
|------|------------------------|---------------|
| 对象新增属性 | ❌ 需 `$set` | ✅ 自动响应 |
| 数组索引修改 | ❌ 需 `$set` | ✅ 自动响应 |
| 数组长度修改 | ❌ 不响应 | ✅ 自动响应 |
| Map/Set | ❌ 不支持 | ✅ 支持 |
| 性能 | 初始化时递归 | 惰性响应 |
| 兼容性 | IE11 支持 | 不支持 IE11 |

## 🔧 实现原理

### 1. Proxy 代理

```typescript
const obj = { count: 0, name: 'Vue' }

const proxy = new Proxy(obj, {
  // 拦截读取
  get(target, key, receiver) {
    console.log(`读取 ${String(key)}`)
    track(target, key)  // 依赖收集
    return Reflect.get(target, key, receiver)
  },
  
  // 拦截设置
  set(target, key, value, receiver) {
    console.log(`设置 ${String(key)} = ${value}`)
    const result = Reflect.set(target, key, value, receiver)
    trigger(target, key)  // 触发更新
    return result
  },
  
  // 拦截删除
  deleteProperty(target, key) {
    console.log(`删除 ${String(key)}`)
    const result = Reflect.deleteProperty(target, key)
    trigger(target, key)
    return result
  }
})

proxy.count  // 读取 count
proxy.count = 1  // 设置 count = 1
```

### 2. 简易响应式实现

```typescript
// 活动副作用
let activeEffect: Function | null = null
// 依赖映射表
const targetMap = new WeakMap<any, Map<string | symbol, Set<Function>>>()

// 收集依赖
function track(target: object, key: string | symbol) {
  if (!activeEffect) return
  
  // 获取目标对象的依赖映射
  let depsMap = targetMap.get(target)
  if (!depsMap) {
    targetMap.set(target, (depsMap = new Map()))
  }
  
  // 获取属性的依赖集合
  let dep = depsMap.get(key)
  if (!dep) {
    depsMap.set(key, (dep = new Set()))
  }
  
  // 收集当前副作用
  dep.add(activeEffect)
}

// 触发更新
function trigger(target: object, key: string | symbol) {
  const depsMap = targetMap.get(target)
  if (!depsMap) return
  
  const dep = depsMap.get(key)
  if (dep) {
    dep.forEach(effect => effect())
  }
}

// 副作用函数
function effect(fn: Function) {
  activeEffect = fn as any
  fn()
  activeEffect = null
}

// 响应式函数
function reactive<T extends object>(target: T): T {
  return new Proxy(target, {
    get(target, key, receiver) {
      track(target, key)
      return Reflect.get(target, key, receiver)
    },
    set(target, key, value, receiver) {
      const result = Reflect.set(target, key, value, receiver)
      trigger(target, key)
      return result
    }
  })
}

// 使用示例
const state = reactive({ count: 0 })

effect(() => {
  console.log('count changed:', state.count)
})

state.count++  // 自动触发: count changed: 1
```

## 📊 依赖收集流程

```
┌─────────────────────────────────────────────────────────────┐
│                       渲染/计算                              │
│  render() { return state.count }                            │
└─────────────────────────────────────────────────────────────┘
                          │
                          ▼
┌─────────────────────────────────────────────────────────────┐
│                    触发 getter                               │
│  Proxy.get() → track(target, key)                          │
└─────────────────────────────────────────────────────────────┘
                          │
                          ▼
┌─────────────────────────────────────────────────────────────┐
│                    依赖收集                                  │
│  targetMap.get(target).get(key).add(activeEffect)          │
│                                                             │
│  targetMap = {                                              │
│    [target]: {                                              │
│      count: Set([effect1, effect2, ...])                   │
│    }                                                        │
│  }                                                          │
└─────────────────────────────────────────────────────────────┘
                          │
                          ▼ (数据变化时)
┌─────────────────────────────────────────────────────────────┐
│                    触发 setter                               │
│  Proxy.set() → trigger(target, key)                        │
└─────────────────────────────────────────────────────────────┘
                          │
                          ▼
┌─────────────────────────────────────────────────────────────┐
│                    执行副作用                                │
│  dep.forEach(effect => effect())                           │
│  → 重新渲染组件                                             │
│  → 重新计算 computed                                        │
└─────────────────────────────────────────────────────────────┘
```

## 🔄 ref 的实现

ref 用于包装基础类型值，通过 `.value` 访问：

```typescript
class RefImpl<T> {
  private _value: T
  private _rawValue: T
  
  public dep?: Set<Function> = undefined
  public readonly __v_isRef = true
  
  constructor(value: T) {
    this._rawValue = value
    // 如果是对象，用 reactive 包装
    this._value = isObject(value) ? reactive(value) : value
  }
  
  get value() {
    // 依赖收集
    trackRefValue(this)
    return this._value
  }
  
  set value(newValue: T) {
    if (hasChanged(newValue, this._rawValue)) {
      this._rawValue = newValue
      this._value = isObject(newValue) ? reactive(newValue) : newValue
      // 触发更新
      triggerRefValue(this)
    }
  }
}

function ref<T>(value: T): Ref<T> {
  return new RefImpl(value)
}
```

### 为什么 ref 需要 `.value`

```typescript
// 基础类型无法被 Proxy 代理
const num = 1
const proxy = new Proxy(num, {})  // ❌ 报错

// 所以需要包装成对象
const ref = {
  _value: 1,
  get value() { return this._value },
  set value(v) { this._value = v }
}
```

## 🔍 响应式工具函数

### 1. isRef / isReactive / isProxy

```typescript
import { ref, reactive, isRef, isReactive, isProxy, isReadonly } from 'vue'

const count = ref(0)
const state = reactive({ name: 'Vue' })

isRef(count)        // true
isReactive(state)   // true
isProxy(state)      // true
isReadonly(state)   // false
```

### 2. unref - 解包 ref

```typescript
import { ref, unref, isRef } from 'vue'

const count = ref(0)

// unref 等价于 isRef(val) ? val.value : val
unref(count)  // 0
unref(1)      // 1

// 实现原理
function unref<T>(ref: T | Ref<T>): T {
  return isRef(ref) ? ref.value : ref
}
```

### 3. toRaw - 获取原始对象

```typescript
import { reactive, toRaw, markRaw } from 'vue'

const state = reactive({ count: 0 })
const raw = toRaw(state)

raw === state  // false
raw.count === state.count  // true

// markRaw - 标记对象永远不会转为响应式
const obj = markRaw({ count: 0 })
const reactiveObj = reactive(obj)  // 不会被代理
```

## 🎯 高级响应式 API

### 1. shallowRef / shallowReactive

```typescript
import { shallowRef, shallowReactive, triggerRef } from 'vue'

// shallowRef - 只有 .value 变化才触发更新
const state = shallowRef({ count: 0 })
state.value.count++  // ❌ 不会触发更新
state.value = { count: 1 }  // ✅ 触发更新

// 手动触发更新
state.value.count++
triggerRef(state)  // ✅ 强制触发

// shallowReactive - 只有根级别是响应式的
const shallow = shallowReactive({
  count: 0,
  nested: {
    value: 1
  }
})
shallow.count++  // ✅ 触发更新
shallow.nested.value++  // ❌ 不会触发更新
```

### 2. customRef - 自定义 ref

```typescript
import { customRef } from 'vue'

// 防抖 ref
function useDebouncedRef<T>(value: T, delay = 200) {
  let timeout: number
  return customRef((track, trigger) => {
    return {
      get() {
        track()  // 收集依赖
        return value
      },
      set(newValue: T) {
        clearTimeout(timeout)
        timeout = setTimeout(() => {
          value = newValue
          trigger()  // 触发更新
        }, delay)
      }
    }
  })
}

// 使用
const text = useDebouncedRef('')
// 输入时延迟 200ms 更新
```

### 3. effectScope - 副作用作用域

```typescript
import { effectScope, ref, watch, computed } from 'vue'

const scope = effectScope()

scope.run(() => {
  const count = ref(0)
  
  // 这些副作用会被收集到 scope 中
  watch(count, () => console.log('changed'))
  const double = computed(() => count.value * 2)
})

// 一次性停止所有副作用
scope.stop()
```

## ⚡ 性能优化

### 1. 惰性响应

```typescript
// Vue 2: 初始化时递归所有属性
Object.defineProperty(obj, 'key', {
  get() { /* ... */ },
  set() { /* ... */ }
})

// Vue 3: 访问时才代理嵌套对象
const state = reactive({
  nested: {
    deep: 1
  }
})
// state.nested.deep 访问时才创建 nested 的代理
```

### 2. 依赖清理

```typescript
// 组件卸载时自动清理依赖
onUnmounted(() => {
  // Vue 自动清理该组件的所有依赖
})

// 手动停止侦听
const stop = watchEffect(() => {})
stop()  // 清理依赖
```

### 3. 避免不必要的响应式

```typescript
import { markRaw } from 'vue'

// 大型静态数据不需要响应式
const bigData = markRaw(largeArray)

// 第三方库实例
const chart = markRaw(new Chart(options))
```

## ⚠️ 常见问题

### 1. 响应式丢失

```typescript
// ❌ 解构 reactive 会丢失响应性
const state = reactive({ count: 0 })
let { count } = state  // count 不是响应式的

// ✅ 使用 toRefs
const { count } = toRefs(state)

// ❌ 重新赋值 reactive
state = { count: 1 }  // 丢失响应性

// ✅ 修改属性
state.count = 1
```

### 2. 数组响应式

```typescript
// ✅ Vue 3 支持直接修改数组索引
const arr = reactive([1, 2, 3])
arr[0] = 10  // 响应式

// ✅ 直接修改长度
arr.length = 1  // 响应式
```

### 3. 对象新增属性

```typescript
// ✅ Vue 3 直接添加新属性
const state = reactive({ name: 'Vue' })
state.count = 0  // 自动响应式
```
