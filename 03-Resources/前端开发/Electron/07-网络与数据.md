## 🌐 网络请求

### 1. net 模块 (推荐)

```javascript
const { net } = require('electron')

// GET 请求
ipcMain.handle('http:get', async (event, url) => {
  return new Promise((resolve, reject) => {
    const request = net.request(url)

    request.on('response', (response) => {
      let data = ''

      response.on('data', (chunk) => {
        data += chunk.toString()
      })

      response.on('end', () => {
        resolve({
          statusCode: response.statusCode,
          headers: response.headers,
          data: data
        })
      })
    })

    request.on('error', (error) => {
      reject(error)
    })

    request.end()
  })
})

// POST 请求
ipcMain.handle('http:post', async (event, url, body, headers = {}) => {
  return new Promise((resolve, reject) => {
    const request = net.request({
      method: 'POST',
      url: url
    })

    // 设置请求头
    Object.entries(headers).forEach(([key, value]) => {
      request.setHeader(key, value)
    })

    request.on('response', (response) => {
      let data = ''
      response.on('data', (chunk) => { data += chunk.toString() })
      response.on('end', () => resolve({
        statusCode: response.statusCode,
        data: data
      }))
    })

    request.on('error', reject)

    // 发送请求体
    request.write(JSON.stringify(body))
    request.end()
  })
})
```

### 4. 拦截网络请求

```javascript
const { session } = require('electron')

// 拦截请求
session.defaultSession.webRequest.onBeforeRequest(
  { urls: ['*://*.example.com/*'] },
  (details, callback) => {
    // 可以修改请求或阻止请求
    if (details.url.includes('ads')) {
      callback({ cancel: true })  // 阻止请求
    } else {
      callback({})
    }
  }
)

// 拦截响应
session.defaultSession.webRequest.onHeadersReceived(
  (details, callback) => {
    // 添加或修改响应头
    callback({
      responseHeaders: {
        ...details.responseHeaders,
        'X-Custom-Header': ['value']
      }
    })
  }
)
```

## 💾 本地存储方案

### 1. electron-store (简单键值存储)

```javascript
// 安装: npm install electron-store

const Store = require('electron-store')
const store = new Store()

// 存储数据
store.set('user.name', 'John')
store.set('user.email', 'john@example.com')
store.set('settings', {
  theme: 'dark',
  fontSize: 14
})

// 读取数据
const name = store.get('user.name')
const settings = store.get('settings')

// 检查是否存在
store.has('user.name')  // true

// 删除数据
store.delete('user.name')

// 清空所有数据
store.clear()

// 获取所有数据
store.store  // 返回所有数据对象

// 存储路径
store.path  // /Users/.../Application Support/MyApp/config.json
```

**封装为 IPC 接口**

```javascript
// main.js
const Store = require('electron-store')
const store = new Store()

ipcMain.handle('store:get', (event, key) => store.get(key))
ipcMain.handle('store:set', (event, key, value) => {
  store.set(key, value)
  return true
})
ipcMain.handle('store:delete', (event, key) => {
  store.delete(key)
  return true
})
ipcMain.handle('store:has', (event, key) => store.has(key))
ipcMain.handle('store:clear', () => {
  store.clear()
  return true
})
```

### 2. SQLite (关系型数据库)

```javascript
// 安装: npm install better-sqlite3

const Database = require('better-sqlite3')
const path = require('path')

// 创建数据库连接
const dbPath = path.join(app.getPath('userData'), 'app.db')
const db = new Database(dbPath)

// 创建表
db.exec(`
  CREATE TABLE IF NOT EXISTS notes (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,
    content TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
  )
`)

// 插入数据
ipcMain.handle('db:insertNote', (event, title, content) => {
  const stmt = db.prepare('INSERT INTO notes (title, content) VALUES (?, ?)')
  const info = stmt.run(title, content)
  return info.lastInsertRowid
})

// 查询数据
ipcMain.handle('db:getNotes', () => {
  const stmt = db.prepare('SELECT * FROM notes ORDER BY created_at DESC')
  return stmt.all()
})

// 更新数据
ipcMain.handle('db:updateNote', (event, id, title, content) => {
  const stmt = db.prepare('UPDATE notes SET title = ?, content = ?, updated_at = CURRENT_TIMESTAMP WHERE id = ?')
  return stmt.run(title, content, id).changes
})

// 删除数据
ipcMain.handle('db:deleteNote', (event, id) => {
  const stmt = db.prepare('DELETE FROM notes WHERE id = ?')
  return stmt.run(id).changes
})

// 搜索
ipcMain.handle('db:searchNotes', (event, keyword) => {
  const stmt = db.prepare('SELECT * FROM notes WHERE title LIKE ? OR content LIKE ?')
  return stmt.all(`%${keyword}%`, `%${keyword}%`)
})

// 关闭数据库
app.on('will-quit', () => {
  db.close()
})
```

### 3. IndexedDB (渲染进程)

```javascript
// 在渲染进程中直接使用 IndexedDB

// 打开数据库
const request = indexedDB.open('MyAppDB', 1)

request.onupgradeneeded = (event) => {
  const db = event.target.result

  // 创建对象存储
  if (!db.objectStoreNames.contains('notes')) {
    const store = db.createObjectStore('notes', { keyPath: 'id', autoIncrement: true })
    store.createIndex('title', 'title', { unique: false })
    store.createIndex('createdAt', 'createdAt', { unique: false })
  }
}

request.onsuccess = (event) => {
  const db = event.target.result

  // 添加数据
  const addNote = (note) => {
    const transaction = db.transaction(['notes'], 'readwrite')
    const store = transaction.objectStore('notes')
    store.add({ ...note, createdAt: new Date() })
  }

  // 获取所有数据
  const getAllNotes = () => {
    return new Promise((resolve, reject) => {
      const transaction = db.transaction(['notes'], 'readonly')
      const store = transaction.objectStore('notes')
      const request = store.getAll()

      request.onsuccess = () => resolve(request.result)
      request.onerror = () => reject(request.error)
    })
  }
}
```

## 📁 文件存储

### 1. JSON 文件

```javascript
const fs = require('fs').promises
const path = require('path')

const dataPath = path.join(app.getPath('userData'), 'data.json')

// 读取 JSON
ipcMain.handle('json:read', async () => {
  try {
    const content = await fs.readFile(dataPath, 'utf-8')
    return JSON.parse(content)
  } catch {
    return {}
  }
})

// 写入 JSON
ipcMain.handle('json:write', async (event, data) => {
  await fs.writeFile(dataPath, JSON.stringify(data, null, 2))
  return true
})
```

### 2. 二进制文件

```javascript
// 保存图片
ipcMain.handle('file:saveImage', async (event, dataURL, filename) => {
  const base64Data = dataURL.replace(/^data:image\/\w+;base64,/, '')
  const buffer = Buffer.from(base64Data, 'base64')

  const imagePath = path.join(app.getPath('pictures'), filename)
  await fs.writeFile(imagePath, buffer)

  return imagePath
})

// 读取图片
ipcMain.handle('file:readImage', async (event, imagePath) => {
  const buffer = await fs.readFile(imagePath)
  return `data:image/png;base64,${buffer.toString('base64')}`
})
```

## 🔄 数据同步

### 1. 简单同步方案

```javascript
// 定期保存到云端
async function syncToCloud() {
  const localData = store.store
  const response = await axios.post('https://api.example.com/sync', {
    data: localData,
    lastSync: store.get('lastSyncTime')
  })

  if (response.data.updated) {
    store.set('lastSyncTime', Date.now())
  }
}

// 定时同步
setInterval(syncToCloud, 5 * 60 * 1000)  // 每 5 分钟
```

### 2. 冲突解决

```javascript
async function syncWithConflictResolution(localData, remoteData) {
  const localTime = localData.updatedAt
  const remoteTime = remoteData.updatedAt

  if (localTime > remoteTime) {
    // 本地更新，推送到远程
    await pushToRemote(localData)
    return localData
  } else if (remoteTime > localTime) {
    // 远程更新，拉取到本地
    store.store = remoteData
    return remoteData
  } else {
    // 无变化
    return localData
  }
}
```

## 📊 存储方案对比

| 方案 | 特点 | 适用场景 | 性能 |
|------|------|----------|------|
| electron-store | 简单键值 | 配置、小数据 | 优秀 |
| SQLite | 关系型 | 结构化数据 | 优秀 |
| IndexedDB | NoSQL | 大量数据 | 良好 |
| localStorage | 简单键值 | 临时数据 | 优秀 |
| JSON 文件 | 灵活 | 可导出数据 | 良好 |
