## 🔍 DevTools 调试

### 1. 打开 DevTools

```javascript
// 自动打开
mainWindow.webContents.openDevTools()

// 只在开发环境打开
if (process.env.NODE_ENV === 'development') {
  mainWindow.webContents.openDevTools()
}

// 分离模式
mainWindow.webContents.openDevTools({ mode: 'detach' })
```


### 3. 性能分析

```javascript
// 录制性能
// 1. 打开 DevTools → Performance
// 2. 点击录制按钮
// 3. 执行操作
// 4. 停止录制

// 使用 Console API 标记
console.time('operation')
// ... 操作
console.timeEnd('operation')

// 使用 Performance API
performance.mark('start')
// ... 操作
performance.mark('end')
performance.measure('operation', 'start', 'end')
```

## 🐛 主进程调试

### 1. Chrome DevTools Protocol

```bash
# 启动时开启调试端口
electron --inspect=5858 .
electron --inspect-brk=5858 .  # 在入口处暂停
```

```javascript
// 或在代码中
const { app } = require('electron')

app.on('ready', () => {
  // 开启调试
  if (process.env.DEBUG) {
    const { BrowserWindow } = require('electron')
    // 主进程调试需要外部工具
  }
})
```

### 2. VS Code 调试配置

```json
// .vscode/launch.json
{
  "version": "0.2.0",
  "configurations": [
    {
      "name": "Debug Main Process",
      "type": "node",
      "request": "launch",
      "cwd": "${workspaceFolder}",
      "runtimeExecutable": "${workspaceFolder}/node_modules/.bin/electron",
      "windows": {
        "runtimeExecutable": "${workspaceFolder}/node_modules/.bin/electron.cmd"
      },
      "args": ["."],
      "outputCapture": "std",
      "sourceMaps": true,
      "resolveSourceMapLocations": [
        "${workspaceFolder}/**",
        "!**/node_modules/**"
      ]
    },
    {
      "name": "Debug Renderer Process",
      "type": "chrome",
      "request": "attach",
      "port": 9222,
      "webRoot": "${workspaceFolder}/src/renderer",
      "timeout": 30000
    }
  ]
}
```

### 3. 使用 electron-log

```bash
npm install electron-log
```

```javascript
// main.js
const log = require('electron-log')

// 配置日志
log.transports.file.level = 'info'
log.transports.console.level = 'debug'

// 日志文件位置
// macOS: ~/Library/Logs/app-name/main.log
// Windows: %USERPROFILE%\AppData\Roaming\app-name\logs\main.log
// Linux: ~/.config/app-name/logs/main.log

// 使用日志
log.info('App starting...')
log.warn('Warning message')
log.error('Error occurred', error)

// 捕获未处理的异常
log.catchErrors({
  showDialog: false,
  onError: (error) => {
    // 自定义错误处理
  }
})
```

## 🧪 自动化测试

### 1. 单元测试 (Jest)

```bash
npm install --save-dev jest @babel/preset-env
```

```javascript
// babel.config.js
module.exports = {
  presets: ['@babel/preset-env']
}

// jest.config.js
module.exports = {
  testEnvironment: 'node',
  testMatch: ['**/__tests__/**/*.js', '**/*.test.js'],
  collectCoverage: true,
  coverageDirectory: 'coverage'
}

// __tests__/calculator.test.js
const { add, subtract } = require('../src/calculator')

describe('Calculator', () => {
  test('adds 1 + 2 to equal 3', () => {
    expect(add(1, 2)).toBe(3)
  })

  test('subtracts 5 - 3 to equal 2', () => {
    expect(subtract(5, 3)).toBe(2)
  })
})
```

### 3. E2E 测试 (Playwright)

```bash
npm install --save-dev @playwright/test
```

```javascript
// playwright.config.ts
import { defineConfig } from '@playwright/test'

export default defineConfig({
  testDir: './e2e',
  fullyParallel: true,
  use: {
    trace: 'on-first-retry',
    video: 'retain-on-failure'
  }
})
```

```javascript
// e2e/app.test.ts
import { _electron as electron, test, expect } from '@playwright/test'

let electronApp
let page

test.beforeAll(async () => {
  electronApp = await electron.launch({
    args: ['.']  // Electron 入口
  })

  page = await electronApp.firstWindow()

  // 等待应用加载
  await page.waitForLoadState('domcontentloaded')
})

test.afterAll(async () => {
  await electronApp.close()
})

test('应用标题正确', async () => {
  const title = await page.title()
  expect(title).toBe('My App')
})

test('窗口可见', async () => {
  const isVisible = await electronApp.evaluate(async ({ BrowserWindow }) => {
    const window = BrowserWindow.getAllWindows()[0]
    return window.isVisible()
  })
  expect(isVisible).toBe(true)
})

test('点击按钮', async () => {
  await page.click('#my-button')
  await expect(page.locator('#result')).toHaveText('Clicked!')
})

test('IPC 通信', async () => {
  // 监听 IPC 事件
  const ipcResult = await electronApp.evaluate(async ({ ipcMain }) => {
    return new Promise(resolve => {
      ipcMain.once('test-event', (event, data) => {
        resolve(data)
      })
    })
  })

  // 在页面中触发 IPC
  await page.evaluate(() => {
    window.electronAPI.send('test-event', { test: true })
  })

  expect(ipcResult).toEqual({ test: true })
})
```
