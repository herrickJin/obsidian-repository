    mainWindow.webContents.session preloadResource(r)
  })
})

### 2. 常见内存泄漏

```javascript
// ❌ 事件监听器泄漏
class Component {
  constructor() {
    window.addEventListener('resize', this.handleResize)
  }

  handleResize = () => {
    // ...
  }

  // 忘记移除监听器！
}

// ✅ 正确清理
class Component {
  constructor() {
    window.addEventListener('resize', this.handleResize)
  }

  destroy() {
    window.removeEventListener('resize', this.handleResize)
  }
}

// ❌ 闭包泄漏
function createHandler() {
  const largeData = new Array(1000000)
  return () => console.log(largeData.length)  // 持有 largeData
}

// ✅ 避免闭包泄漏
function createHandler() {
  const length = 1000000  // 只保存需要的值
  return () => console.log(length)
}

// ❌ 定时器泄漏
setInterval(() => {
  // 如果组件销毁，定时器仍在运行
}, 1000)

// ✅ 清理定时器
class Component {
  constructor() {
    this.timer = setInterval(() => {}, 1000)
  }

  destroy() {
    clearInterval(this.timer)
  }
}
```

### 3. 节流和防抖

```javascript
// 防抖 - 等待停止后再执行
function debounce(fn, delay) {
  let timer
  return (...args) => {
    clearTimeout(timer)
    timer = setTimeout(() => fn(...args), delay)
  }
}

// 节流 - 限制执行频率
function throttle(fn, limit) {
  let inThrottle
  return (...args) => {
    if (!inThrottle) {
      fn(...args)
      inThrottle = true
      setTimeout(() => inThrottle = false, limit)
    }
  }
}

// 使用
const handleResize = debounce(() => {
  console.log('Resized')
}, 200)

const handleScroll = throttle(() => {
  console.log('Scrolling')
}, 100)
```

## 🎨 渲染优化

### 1. 虚拟列表

```javascript
// 只渲染可见项
class VirtualList {
  constructor(options) {
    this.container = options.container
    this.itemHeight = options.itemHeight
    this.renderItem = options.renderItem
    this.data = options.data || []

    this.visibleCount = Math.ceil(this.container.clientHeight / this.itemHeight) + 2
    this.startIndex = 0

    this.init()
  }

  init() {
    this.wrapper = document.createElement('div')
    this.wrapper.style.position = 'relative'
    this.wrapper.style.height = `${this.data.length * this.itemHeight}px`
    this.container.appendChild(this.wrapper)

    this.render()
    this.container.addEventListener('scroll', () => this.onScroll())
  }

  onScroll() {
    const scrollTop = this.container.scrollTop
    this.startIndex = Math.floor(scrollTop / this.itemHeight)
    this.render()
  }

  render() {
    const endIndex = Math.min(this.startIndex + this.visibleCount, this.data.length)

    // 清除旧元素
    this.wrapper.innerHTML = ''

    // 只渲染可见项
    for (let i = this.startIndex; i < endIndex; i++) {
      const item = this.renderItem(this.data[i], i)
      item.style.position = 'absolute'
      item.style.top = `${i * this.itemHeight}px`
      item.style.height = `${this.itemHeight}px`
      this.wrapper.appendChild(item)
    }
  }
}
```
