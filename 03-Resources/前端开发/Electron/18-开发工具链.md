## ⚡ electron-vite

### 1. 项目创建

```bash
# 创建新项目
npm create @electron-vite/create@latest

# 或手动安装
npm install --save-dev electron-vite vite
```

### 2. 项目结构

```
my-app/
├── electron.vite.config.js
├── package.json
├── src/
│   ├── main/              # 主进程
│   │   └── index.js
│   ├── preload/           # 预加载脚本
│   │   └── index.js
│   └── renderer/          # 渲染进程
│       ├── index.html
│       ├── src/
│       │   ├── main.js
│       │   └── App.vue
│       └── package.json
└── resources/             # 资源文件
    └── icon.png
```

### 3. 配置文件

```javascript
// electron.vite.config.js
import { defineConfig } from 'electron-vite'
import vue from '@vitejs/plugin-vue'
import { resolve } from 'path'

export default defineConfig({
  main: {
    // 主进程配置
    build: {
      rollupOptions: {
        input: {
          index: resolve(__dirname, 'src/main/index.js')
        }
      }
    }
  },
  preload: {
    // 预加载脚本配置
    build: {
      rollupOptions: {
        input: {
          index: resolve(__dirname, 'src/preload/index.js')
        }
      }
    }
  },
  renderer: {
    // 渲染进程配置
    root: '.',
    build: {
      rollupOptions: {
        input: {
          index: resolve(__dirname, 'src/renderer/index.html')
        }
      }
    },
    plugins: [vue()],
    resolve: {
      alias: {
        '@': resolve(__dirname, 'src/renderer/src')
      }
    }
  }
})
```

### 4. package.json 脚本

```json
{
```

## 🔥 Vite 集成

### 1. 手动集成 Vite

```javascript
// vite.config.js
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'

export default defineConfig({
  plugins: [vue()],
  base: './',
  build: {
    outDir: 'dist/renderer',
    emptyOutDir: true
  },
  server: {
    port: 5173
  }
})
```

### 2. 主进程开发脚本

```javascript
// scripts/dev.js
const { spawn } = require('child_process')
const { createServer } = require('vite')
const electron = require('electron')
const path = require('path')

async function dev() {
  // 启动 Vite 开发服务器
  const viteServer = await createServer({
    configFile: path.join(__dirname, '../vite.config.js')
  })
  await viteServer.listen()

  const viteUrl = `http://localhost:${viteServer.config.server.port}`

  // 启动 Electron
  const electronProcess = spawn(electron, ['.'], {
    env: {
      ...process.env,
      VITE_DEV_SERVER_URL: viteUrl
    }
  })

  electronProcess.stdout.on('data', (data) => {
    console.log(`Electron: ${data}`)
  })

  electronProcess.stderr.on('data', (data) => {
    console.error(`Electron Error: ${data}`)
  })
}

dev()
```

### 3. 主进程加载开发服务器

```javascript
// main.js
const { app, BrowserWindow } = require('electron')
const path = require('path')

function createWindow() {
  const win = new BrowserWindow({
    width: 800,
    height: 600,
    webPreferences: {
      preload: path.join(__dirname, 'preload.js')
    }
  })

  // 开发环境加载 Vite 服务器
  if (process.env.VITE_DEV_SERVER_URL) {
    win.loadURL(process.env.VITE_DEV_SERVER_URL)
    win.webContents.openDevTools()
  } else {
    // 生产环境加载构建产物
    win.loadFile(path.join(__dirname, '../dist/renderer/index.html'))
  }
}

app.whenReady().then(createWindow)
```

## 🔄 热重载

### 1. electron-reload

```bash
npm install --save-dev electron-reload
```

```javascript
// main.js
const path = require('path')

// 开发环境热重载
if (process.env.NODE_ENV === 'development') {
  require('electron-reload')(
    path.join(__dirname, '..'),
    {
      electron: require(`${__dirname}/../node_modules/electron`),
      hardResetMethod: 'exit'
    }
  )
}
```

### 2. 自定义热重载

```javascript
// scripts/dev.js
const chokidar = require('chokidar')
const { spawn } = require('child_process')

let electronProcess = null

function startElectron() {
  if (electronProcess) {
    electronProcess.kill()
  }

  electronProcess = spawn('electron', ['.'], {
    env: process.env,
    stdio: 'inherit'
  })
}

// 监听主进程文件变化
const watcher = chokidar.watch([
  'src/main/**/*',
  'src/preload/**/*'
])

watcher.on('change', () => {
  console.log('Main process changed, restarting...')
  startElectron()
})

startElectron()
```

### 3. 渲染进程热更新 (HMR)

```javascript
// Vite 自带 HMR，无需额外配置
// 只需要确保加载的是开发服务器 URL

// 在渲染进程中可以监听 HMR 更新
if (import.meta.hot) {
  import.meta.hot.accept('./App.vue', (newModule) => {
    // 处理更新
  })
}
```

### 2. 类型声明

```typescript
// src/preload/index.d.ts
import { ElectronAPI } from '@electron-toolkit/preload'

declare global {
  interface Window {
    electron: ElectronAPI
    api: {
      // 自定义 API 类型
      getUser: () => Promise<User>
      saveSettings: (settings: Settings) => Promise<void>

  }
}

  id: number
  name: string
  email: string

  theme: 'light' | 'dark'
  language: string
```

### 3. 主进程 TypeScript

```typescript
// src/main/index.ts
import { app, BrowserWindow, ipcMain } from 'electron'
import { join } from 'path'

function createWindow(): void {
  const win = new BrowserWindow({
    width: 800,
    height: 600,
    webPreferences: {
      preload: join(__dirname, '../preload/index.js'),
      contextIsolation: true,
      nodeIntegration: false
    }
  })

  // 开发环境
  if (process.env.NODE_ENV === 'development') {
    win.loadURL('http://localhost:5173')
    win.webContents.openDevTools()
  } else {
    win.loadFile(join(__dirname, '../renderer/index.html'))
  }
}

// IPC 处理器
ipcMain.handle('get-user', async (): Promise<User> => {
  return {
    id: 1,
    name: 'John',
    email: 'john@example.com'
  }
})

app.whenReady().then(createWindow)
```
