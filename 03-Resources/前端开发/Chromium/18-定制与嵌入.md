### Content API 接口

```cpp
// 核心 Content API 接口

// ContentClient - 全局配置
class ContentClient {
 public:
  virtual std::string GetUserAgent();
  virtual base::string16 GetLocalizedString(int message_id);
  virtual base::StringPiece GetDataResource(int resource_id);
  virtual GURL GetWebUIURL(const std::string& host);
};

// ContentBrowserClient - 浏览器进程配置
class ContentBrowserClient {
 public:
  // 创建 WebContents
  virtual std::unique_ptr<WebContents> CreateWebContents(...);

  // 资源请求处理
  virtual void ResourceDispatcherHostCreated(...);

  // 进程创建
  virtual void RenderProcessHostCreated(...);

  // 导航控制
  virtual bool ShouldAllowNavigation(...);

  // 权限管理
  virtual void RequestPermission(...);
};

// ContentRendererClient - 渲染进程配置
class ContentRendererClient {
 public:
  // 渲染线程启动
  virtual void RenderThreadStarted();

  // 视图创建
  virtual void RenderViewCreated(...);

  // 脚本注入
  virtual void DidCreateScriptContext(...);

  // 导航事件
  virtual void WillSendRequest(...);
};

// ContentUtilityClient - 工具进程配置
class ContentUtilityClient {
 public:
  virtual void UtilityThreadStarted();
  virtual void OnBindService(...);
};
```

### 最小嵌入示例

```cpp
// 最小 Chromium 嵌入示例

#include "content/public/app/content_main.h"
#include "content/public/app/content_main_delegate.h"

class MyContentMainDelegate : public content::ContentMainDelegate {
 public:
  ContentBrowserClient* CreateContentBrowserClient() override {
    return &browser_client_;
  }

  ContentRendererClient* CreateContentRendererClient() override {
    return &renderer_client_;
  }

 private:
  MyContentBrowserClient browser_client_;
  MyContentRendererClient renderer_client_;
};

// 入口点
#if defined(OS_WIN)
int APIENTRY wWinMain(HINSTANCE instance, HINSTANCE, wchar_t*, int) {
  MyContentMainDelegate delegate;
  content::ContentMainParams params(&delegate);
  params.instance = instance;
  return content::ContentMain(std::move(params));
}
#elif defined(OS_MACOSX)
int main(int argc, const char** argv) {
  MyContentMainDelegate delegate;
  content::ContentMainParams params(&delegate);
  params.argc = argc;
  params.argv = argv;
  return content::ContentMain(std::move(params));
}
#endif
```

### CEF 基本使用

```cpp
// CEF C++ 示例

#include "include/cef_app.h"
#include "include/cef_client.h"
#include "include/cef_browser.h"

// 自定义客户端处理器
class MyClient : public CefClient,
                 public CefLifeSpanHandler,
                 public CefLoadHandler {
 public:
  // CefLifeSpanHandler
  void OnAfterCreated(CefRefPtr<CefBrowser> browser) override {
    // 浏览器创建完成
  }

  bool DoClose(CefRefPtr<CefBrowser> browser) override {
    return false;  // 允许关闭
  }

  void OnBeforeClose(CefRefPtr<CefBrowser> browser) override {
    // 浏览器关闭
  }

  // CefLoadHandler
  void OnLoadStart(CefRefPtr<CefBrowser> browser,
                   CefRefPtr<CefFrame> frame) override {
    // 开始加载
  }

  void OnLoadEnd(CefRefPtr<CefBrowser> browser,
                 CefRefPtr<CefFrame> frame,
                 int httpStatusCode) override {
    // 加载完成
  }

  void OnLoadError(CefRefPtr<CefBrowser> browser,
                   CefRefPtr<CefFrame> frame,
                   ErrorCode errorCode,
                   const CefString& errorText,
                   const CefString& failedUrl) override {
    // 加载错误
  }

  // 获取处理器
  CefRefPtr<CefLifeSpanHandler> GetLifeSpanHandler() override {
    return this;
  }
  CefRefPtr<CefLoadHandler> GetLoadHandler() override {
    return this;
  }

  IMPLEMENT_REFCOUNTING(MyClient);
};

// 创建浏览器
void CreateBrowser(CefRefPtr<CefClient> client) {
  CefWindowInfo window_info;
  window_info.SetAsWindowless(nullptr);  // 无窗口模式

  CefBrowserSettings browser_settings;

  CefBrowserHost::CreateBrowser(
      window_info,
      client,
      "https://example.com",
      browser_settings,
      nullptr,
      nullptr);
}

// 主函数
int main(int argc, char* argv[]) {
  // 初始化 CEF
  CefMainArgs main_args(argc, argv);
  CefRefPtr<CefApp> app;

  // 处理子进程
  int exit_code = CefExecuteProcess(main_args, app, nullptr);
  if (exit_code >= 0) {
    return exit_code;
  }

  // 初始化设置
  CefSettings settings;
  settings.no_sandbox = true;

  // 初始化 CEF
  CefInitialize(main_args, settings, app, nullptr);

  // 运行消息循环
  CefRunMessageLoop();

  // 清理
  CefShutdown();

  return 0;
}
```

### CEF 与 JavaScript 交互

```cpp
// C++ 调用 JavaScript
void CallJavaScript(CefRefPtr<CefBrowser> browser) {
  CefRefPtr<CefFrame> frame = browser->GetMainFrame();

  // 执行 JavaScript
  frame->ExecuteJavaScript(
      "alert('Hello from C++!')",
      frame->GetURL(),
      0);
}

// JavaScript 调用 C++ (通过 V8 扩展)
class MyV8Handler : public CefV8Handler {
 public:
  bool Execute(const CefString& name,
               CefRefPtr<CefV8Value> object,
               const CefV8ValueList& arguments,
               CefRefPtr<CefV8Value>& retval,
               CefString& exception) override {
    if (name == "myFunction") {
      // 处理调用
      retval = CefV8Value::CreateString("Hello from C++!");
      return true;
    }
    return false;
  }

  IMPLEMENT_REFCOUNTING(MyV8Handler);
};

// 注册 V8 扩展
void RegisterExtension() {
  std::string code = R"(
    var myapp;
    if (!myapp)
      myapp = {};

    (function() {
      myapp.myFunction = function() {
        native function myFunction();
        return myFunction();
      };
    })();
  )";

  CefRegisterExtension("v8/myapp", code, new MyV8Handler());
}
```

### Electron 与 Chromium 的关系

```cpp
// Electron 扩展 Chromium 的方式

// 1. 主进程 (Main Process)
// - 扩展 ContentBrowserClient
// - 添加 Node.js 集成
// - 提供 Electron API

class ElectronBrowserClient : public content::ContentBrowserClient {
 public:
  // 扩展 API
  void OverrideWebkitPrefs(content::RenderViewHost* rvh,
                           blink::web_pref::WebPreferences* prefs) override {
    // 自定义 WebKit 配置
    prefs->node_integration_in_worker = true;
  }
};

// 2. 渲染进程 (Renderer Process)
// - 扩展 ContentRendererClient
// - 注入 preload 脚本
// - 提供 contextBridge API

class ElectronRendererClient : public content::ContentRendererClient {
 public:
  void RenderViewCreated(content::RenderView* render_view) override {
    // 注入 preload 脚本
    if (ShouldInjectPreload()) {
      InjectPreloadScript(render_view);
    }
  }

  void DidCreateScriptContext(
      v8::Handle<v8::Context> context,
      content::RenderFrame* render_frame) override {
    // 设置 Node.js 环境
    if (node_integration_) {
      node::Environment::Create(context);
    }
  }
};
```

### 添加自定义 API

```cpp
// 示例: 添加自定义 Web API

// 1. 定义 IDL 接口
// third_party/blink/renderer/modules/custom/custom_api.idl
interface CustomAPI {
  void doSomething();
  Promise<string> getData();
};

// 2. 实现 C++ 类
// third_party/blink/renderer/modules/custom/custom_api.h
class CustomAPI : public ScriptWrappable {
  DEFINE_WRAPPERTYPEINFO();
 public:
  static CustomAPI* Create();

  void doSomething();
  ScriptPromise getData(ScriptState*);

 private:
  CustomAPI();
};

// 3. 注册到 Window
// third_party/blink/renderer/core/frame/window.idl
[RuntimeEnabled=CustomAPI] attribute CustomAPI customAPI;

// 4. 构建配置
// third_party/blink/renderer/modules/custom/BUILD.gn
blink_modules_sources("custom") {
  sources = [
    "custom_api.cc",
    "custom_api.h",
  ]
}
```

### 编译自定义版本

```bash
# 1. 修改源码后重新生成构建文件

  # 启用自定义功能

# 3. 打包 (Linux)
ninja -C out/Custom chrome/installer/linux:stable

# 4. 打包 (macOS)
ninja -C out/Custom chrome/installer/mac:stable
```
