# 问题排查指南

> WebRTC 问题排查的系统方法，从常见问题到解决方案

## 📌 排查思路

```
┌─────────────────────────────────────────────────────────────┐
│                    问题排查流程                              │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  1. 确认问题现象                                            │
│       ↓                                                     │
│  2. 检查基础环境 (HTTPS/权限/浏览器支持)                     │
│       ↓                                                     │
│  3. 检查连接状态 (ICE/DTLS/SRTP)                            │
│       ↓                                                     │
│  4. 检查网络质量 (RTT/丢包/带宽)                            │
│       ↓                                                     │
│  5. 检查媒体质量 (编解码/帧率/分辨率)                        │
│       ↓                                                     │
│  6. 定位根本原因                                            │
│       ↓                                                     │
│  7. 应用解决方案                                            │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

---

## 🚨 常见问题分类

### 1. 连接问题

| 问题 | 现象 | 可能原因 |
|------|------|----------|
| 无法获取媒体 | getUserMedia 失败 | 权限/设备/HTTPS |
| ICE 失败 | iceConnectionState = failed | NAT/防火墙/STUN |
| DTLS 失败 | 连接卡住 | 证书/指纹 |
| 无法连接 | 一直 checking | 信令/网络 |

### 2. 质量问题

| 问题 | 现象 | 可能原因 |
|------|------|----------|
| 视频卡顿 | 画面冻结 | 丢包/带宽不足 |
| 音频断续 | 声音卡顿 | 抖动/丢包 |
| 画面模糊 | 分辨率低 | 带宽限制 |
| 延迟高 | 唇音不同步 | 网络延迟/缓冲 |

### 3. 功能问题

| 问题 | 现象 | 可能原因 |
|------|------|----------|
| 无声音 | 只有视频 | 音频轨道/编解码 |
| 无画面 | 只有音频 | 视频轨道/编解码 |
| 单通 | 只有单向 | 防火墙/NAT |

## 🔧 详细排查步骤

### 问题 1: getUserMedia 失败

```javascript
// 现象
// NotFoundError / NotAllowedError / NotReadableError

// 排查步骤
async function debugGetUserMedia() {
  // 1. 检查 HTTPS
  if (location.protocol !== 'https:' && location.hostname !== 'localhost') {
    console.error('❌ 需要 HTTPS 环境');
    return;
  }

  // 2. 检查 API 支持
  if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
    console.error('❌ 浏览器不支持 getUserMedia');
    return;
  }

  // 3. 检查权限状态
  try {
    const camera = await navigator.permissions.query({ name: 'camera' });
    const mic = await navigator.permissions.query({ name: 'microphone' });
    console.log('摄像头权限:', camera.state);
    console.log('麦克风权限:', mic.state);
  } catch (e) {
    console.log('权限 API 不支持');
  }

  // 4. 列出设备
  const devices = await navigator.mediaDevices.enumerateDevices();
  console.log('设备列表:', devices);

  // 5. 尝试获取
  try {
    const stream = await navigator.mediaDevices.getUserMedia({
      video: true,
      audio: true
    });
    console.log('✅ 获取成功');
    stream.getTracks().forEach(t => t.stop());
  } catch (error) {
    console.error('❌ 获取失败:', error.name, error.message);

    switch (error.name) {
      case 'NotAllowedError':
        console.log('解决方案: 用户拒绝了权限，需要引导用户允许');
        break;
      case 'NotFoundError':
        console.log('解决方案: 没有找到设备，检查设备连接');
        break;
      case 'NotReadableError':
        console.log('解决方案: 设备被占用，关闭其他应用');
        break;
      case 'OverconstrainedError':
        console.log('解决方案: 约束条件无法满足，降低要求');
        break;
    }
  }
}
```

### 问题 2: ICE 连接失败

```javascript
// 现象: iceConnectionState = 'failed'

async function debugICEFailure(pc) {
  console.log('=== ICE 调试开始 ===');

  // 1. 检查 ICE 配置
  const config = pc.getConfiguration();
  console.log('ICE 服务器:', config.iceServers);

  if (!config.iceServers || config.iceServers.length === 0) {
    console.error('❌ 没有配置 ICE 服务器');
    return;
  }

  // 2. 检查候选收集
  const stats = await pc.getStats();
  const candidates = { local: [], remote: [] };

  stats.forEach(report => {
    if (report.type === 'local-candidate') {
      candidates.local.push(report);
    }
    if (report.type === 'remote-candidate') {
      candidates.remote.push(report);
    }
  });

  console.log('本地候选:', candidates.local);
  console.log('远程候选:', candidates.remote);

  // 3. 诊断
  const hasHost = candidates.local.some(c => c.candidateType === 'host');
  const hasSrflx = candidates.local.some(c => c.candidateType === 'srflx');
  const hasRelay = candidates.local.some(c => c.candidateType === 'relay');

  if (!hasHost) {
    console.warn('⚠️ 没有 host 候选，可能是网络接口问题');
  }

  if (!hasSrflx && !hasRelay) {
    console.error('❌ 没有 srflx/relay 候选，STUN/TURN 可能不可用');
    console.log('解决方案: 检查 STUN/TURN 服务器配置和可达性');
  }

  if (candidates.remote.length === 0) {
    console.error('❌ 没有收到远程候选');
    console.log('解决方案: 检查信令服务器是否正常传递候选');
  }

  // 4. 检查候选对
  stats.forEach(report => {
    if (report.type === 'candidate-pair') {
      console.log('候选对状态:', report.state, report.nominated ? '(nominated)' : '');
    }
  });

  console.log('=== ICE 调试结束 ===');
}

// 监听 ICE 状态
pc.oniceconnectionstatechange = () => {
  console.log('ICE 状态:', pc.iceConnectionState);

  if (pc.iceConnectionState === 'failed') {
    debugICEFailure(pc);
  }
};
```

### 问题 3: 视频卡顿/冻结

```javascript
// 现象: 画面卡住不动

async function debugVideoFreeze(pc) {
  console.log('=== 视频卡顿调试开始 ===');

  const stats = await pc.getStats();

  stats.forEach(report => {
    if (report.type === 'inbound-rtp' && report.kind === 'video') {
      console.log('视频接收统计:', {
        framesReceived: report.framesReceived,
        framesDecoded: report.framesDecoded,
        framesDropped: report.framesDropped,
        framesPerSecond: report.framesPerSecond,
        packetsLost: report.packetsLost,
        packetsReceived: report.packetsReceived,
        jitter: report.jitter,
        firCount: report.firCount,
        pliCount: report.pliCount,
        nackCount: report.nackCount
      });

      // 诊断
      const lossRate = report.packetsLost / (report.packetsReceived + report.packetsLost);
      console.log('丢包率:', (lossRate * 100).toFixed(2) + '%');

      if (lossRate > 0.1) {
        console.error('❌ 丢包率过高 (>10%)');
        console.log('解决方案: 检查网络质量，启用 FEC');
      }

      if (report.framesDropped > report.framesDecoded * 0.1) {
        console.warn('⚠️ 丢帧率较高');
        console.log('解决方案: 可能解码性能不足，降低分辨率');
      }

      if (report.pliCount > 10) {
        console.warn('⚠️ 频繁请求关键帧');
        console.log('解决方案: 增加关键帧间隔，检查编码器');
      }
    }
  });

  // 检查网络
  stats.forEach(report => {
    if (report.type === 'candidate-pair' && report.state === 'succeeded') {
      console.log('网络统计:', {
        rtt: (report.currentRoundTripTime * 1000).toFixed(0) + 'ms',
        availableBitrate: ((report.availableOutgoingBitrate || 0) / 1000000).toFixed(2) + 'Mbps'
      });

      if (report.currentRoundTripTime > 0.2) {
        console.warn('⚠️ RTT 过高 (>200ms)');
      }
    }
  });

  console.log('=== 视频卡顿调试结束 ===');
}
```

### 问题 4: 音频问题

```javascript
// 现象: 无声音/断续/回声

async function debugAudioIssues(pc) {
  console.log('=== 音频调试开始 ===');

  const stats = await pc.getStats();

  stats.forEach(report => {
    if (report.type === 'inbound-rtp' && report.kind === 'audio') {
      console.log('音频接收统计:', {
        packetsReceived: report.packetsReceived,
        packetsLost: report.packetsLost,
        jitter: (report.jitter * 1000).toFixed(2) + 'ms'
      });

      const lossRate = report.packetsLost / (report.packetsReceived + report.packetsLost);
      if (lossRate > 0.05) {
        console.warn('⚠️ 音频丢包率过高');
      }

      if (report.jitter > 0.03) {
        console.warn('⚠️ 音频抖动过大');
      }
    }

    if (report.type === 'outbound-rtp' && report.kind === 'audio') {
      console.log('音频发送统计:', {
        packetsSent: report.packetsSent,
        bytesSent: report.bytesSent
      });
    }
  });

  // 检查音频轨道
  const senders = pc.getSenders();
  for (const sender of senders) {
    if (sender.track?.kind === 'audio') {
      console.log('音频轨道:', {
        enabled: sender.track.enabled,
        muted: sender.track.muted,
        readyState: sender.track.readyState
      });

      if (!sender.track.enabled) {
        console.warn('⚠️ 音频轨道已禁用');
      }
      if (sender.track.muted) {
        console.warn('⚠️ 音频轨道已静音');
      }
    }
  }

  console.log('=== 音频调试结束 ===');
}
```

### 问题 5: 延迟过高

```javascript
// 现象: 唇音不同步，延迟明显

async function debugHighLatency(pc) {
  console.log('=== 延迟调试开始 ===');

  const stats = await pc.getStats();
  let totalLatency = 0;

  // 1. 网络延迟
  stats.forEach(report => {
    if (report.type === 'candidate-pair' && report.state === 'succeeded') {
      const rtt = report.currentRoundTripTime * 1000;
      console.log('RTT:', rtt.toFixed(0) + 'ms');
      totalLatency += rtt / 2; // 单向延迟
    }
  });

  // 2. 抖动缓冲延迟
  stats.forEach(report => {
    if (report.type === 'inbound-rtp') {
      if (report.jitterBufferDelay && report.jitterBufferEmittedCount) {
        const jbDelay = (report.jitterBufferDelay / report.jitterBufferEmittedCount) * 1000;
        console.log(report.kind, '抖动缓冲延迟:', jbDelay.toFixed(0) + 'ms');
        totalLatency += jbDelay;
      }
    }
  });

  // 3. 估算总延迟
  totalLatency += 30; // 编码延迟估算
  totalLatency += 20; // 解码延迟估算

  console.log('估算总延迟:', totalLatency.toFixed(0) + 'ms');

  if (totalLatency > 300) {
    console.warn('⚠️ 延迟过高 (>300ms)');
    console.log('解决方案:');
    console.log('  1. 减少抖动缓冲大小');
    console.log('  2. 降低分辨率/帧率');
    console.log('  3. 使用更近的服务器');
    console.log('  4. 检查网络路由');
  }

  console.log('=== 延迟调试结束 ===');
}
```

## 🛠️ 综合诊断工具

```javascript
class WebRTCDiagnostics {
  constructor(pc) {
    this.pc = pc;
    this.issues = [];
  }

  async runAllChecks() {
    console.log('🔍 开始综合诊断...\n');

    this.issues = [];

    await this.checkConnection();
    await this.checkNetwork();
    await this.checkMedia();
    await this.checkQuality();

    this.printReport();

    return this.issues;
  }

  async checkConnection() {
    // 检查连接状态
    if (this.pc.connectionState === 'failed') {
      this.issues.push({
        severity: 'error',
        category: 'connection',
        message: '连接失败'
      });
    }

    if (this.pc.iceConnectionState === 'failed') {
      this.issues.push({
        severity: 'error',
        category: 'connection',
        message: 'ICE 连接失败'
      });
    }

    if (this.pc.iceConnectionState === 'disconnected') {
      this.issues.push({
        severity: 'warning',
        category: 'connection',
        message: 'ICE 连接断开'
      });
    }
  }

  async checkNetwork() {
    const stats = await this.pc.getStats();

    stats.forEach(report => {
      if (report.type === 'candidate-pair' && report.state === 'succeeded') {
        // RTT
        if (report.currentRoundTripTime > 0.3) {
          this.issues.push({
            severity: 'warning',
            category: 'network',
            message: `高延迟: ${(report.currentRoundTripTime * 1000).toFixed(0)}ms`
          });
        }

        // 带宽
        if (report.availableOutgoingBitrate < 100000) {
          this.issues.push({
            severity: 'warning',
            category: 'network',
            message: `低带宽: ${(report.availableOutgoingBitrate / 1000).toFixed(0)}kbps`
          });
        }
      }
    });
  }

  async checkMedia() {
    const stats = await this.pc.getStats();

    stats.forEach(report => {
      if (report.type === 'inbound-rtp') {
        const lossRate = report.packetsLost / (report.packetsReceived + report.packetsLost || 1);

        if (lossRate > 0.1) {
          this.issues.push({
            severity: 'error',
            category: 'media',
            message: `${report.kind} 高丢包: ${(lossRate * 100).toFixed(1)}%`
          });
        } else if (lossRate > 0.05) {
          this.issues.push({
            severity: 'warning',
            category: 'media',
            message: `${report.kind} 中等丢包: ${(lossRate * 100).toFixed(1)}%`
          });
        }

        if (report.jitter > 0.05) {
          this.issues.push({
            severity: 'warning',
            category: 'media',
            message: `${report.kind} 高抖动: ${(report.jitter * 1000).toFixed(0)}ms`
          });
        }
      }
    });
  }

  async checkQuality() {
    const stats = await this.pc.getStats();

    stats.forEach(report => {
      if (report.type === 'outbound-rtp' && report.kind === 'video') {
        if (report.qualityLimitationReason !== 'none') {
          this.issues.push({
            severity: 'info',
            category: 'quality',
            message: `质量受限: ${report.qualityLimitationReason}`
          });
        }

        if (report.framesPerSecond < 15) {
          this.issues.push({
            severity: 'warning',
            category: 'quality',
            message: `低帧率: ${report.framesPerSecond}fps`
          });
        }
      }
    });
  }

  printReport() {
    console.log('\n📋 诊断报告');
    console.log('='.repeat(50));

    if (this.issues.length === 0) {
      console.log('✅ 未发现问题');
      return;
    }

    const errors = this.issues.filter(i => i.severity === 'error');
    const warnings = this.issues.filter(i => i.severity === 'warning');
    const infos = this.issues.filter(i => i.severity === 'info');

    if (errors.length > 0) {
      console.log('\n❌ 错误:');
      errors.forEach(i => console.log(`   [${i.category}] ${i.message}`));
    }

    if (warnings.length > 0) {
      console.log('\n⚠️  警告:');
      warnings.forEach(i => console.log(`   [${i.category}] ${i.message}`));
    }

    if (infos.length > 0) {
      console.log('\nℹ️  信息:');
      infos.forEach(i => console.log(`   [${i.category}] ${i.message}`));
    }

    console.log('\n' + '='.repeat(50));
  }
}

// 使用
const diagnostics = new WebRTCDiagnostics(pc);
await diagnostics.runAllChecks();
```
