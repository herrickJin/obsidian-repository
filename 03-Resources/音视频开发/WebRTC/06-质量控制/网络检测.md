# 网络检测

> 网络检测是评估网络质量、预测带宽、指导自适应调整的基础

- 检测网络带宽
- 测量网络延迟
- 评估丢包情况

## 📊 带宽检测

### 主动探测

```javascript
class BandwidthProbe {
  constructor() {
    this.probeSize = 10000; // 探测包大小
    this.probeCount = 10;
  }

  async probeUpstream() {
    const results = [];

    for (let i = 0; i < this.probeCount; i++) {
      const start = performance.now();

      // 发送探测包（通过 DataChannel）
      await this.sendProbePacket(this.probeSize);

      const end = performance.now();
      results.push({
        duration: end - start,
        size: this.probeSize
      });

      // 间隔
      await new Promise(r => setTimeout(r, 100));
    }

    // 计算带宽
    return this.calculateBandwidth(results);
  }

  calculateBandwidth(results) {
    const totalBytes = results.reduce((sum, r) => sum + r.size, 0);
    const totalMs = results.reduce((sum, r) => sum + r.duration, 0);

    // bps = bytes * 8 / seconds
    return (totalBytes * 8) / (totalMs / 1000);
  }

  async sendProbePacket(size) {
    // 通过已建立的 DataChannel 发送
    const data = new ArrayBuffer(size);
    this.dataChannel.send(data);
  }
}
```

### 被动估计

```javascript
class PassiveBandwidthEstimator {
  constructor() {
    this.samples = [];
    this.maxSamples = 100;
  }

  // 记录接收/发送样本
  recordSample(bytes, timestamp) {
    this.samples.push({ bytes, timestamp });

    if (this.samples.length > this.maxSamples) {
      this.samples.shift();
    }
  }

  // 计算带宽
  estimate(windowMs = 1000) {
    if (this.samples.length < 2) return 0;

    const now = performance.now();
    const cutoff = now - windowMs;

    const recentSamples = this.samples.filter(s => s.timestamp >= cutoff);

    if (recentSamples.length < 2) return 0;

    const totalBytes = recentSamples.reduce((sum, s) => sum + s.bytes, 0);
    const duration = (now - recentSamples[0].timestamp) / 1000;

    return (totalBytes * 8) / duration; // bps
  }
}
```

### WebRTC getStats 获取带宽

```javascript
async function getBandwidthStats(pc) {
  const stats = await pc.getStats();
  let bandwidth = {
    outgoing: 0,
    incoming: 0,
    rtt: 0
  };

  stats.forEach(report => {
    if (report.type === 'candidate-pair' && report.state === 'succeeded') {
      bandwidth.outgoing = report.availableOutgoingBitrate || 0;
      bandwidth.incoming = report.availableIncomingBitrate || 0;
      bandwidth.rtt = report.currentRoundTripTime || 0;
    }
  });

  return bandwidth;
}
```

## ⏱️ 延迟检测

### RTT 测量

```javascript
class RTTMeasurer {
  constructor() {
    this.rttSamples = [];
    this.maxSamples = 30;
  }

  // 通过 Sender Report 测量
  recordRTT(rtt) {
    this.rttSamples.push({
      value: rtt,
      timestamp: Date.now()
    });

    if (this.rttSamples.length > this.maxSamples) {
      this.rttSamples.shift();
    }
  }

  // 获取统计
  getStats() {
    if (this.rttSamples.length === 0) return null;

    const values = this.rttSamples.map(s => s.value);
    const sum = values.reduce((a, b) => a + b, 0);

    return {
      current: values[values.length - 1],
      average: sum / values.length,
      min: Math.min(...values),
      max: Math.max(...values),
      // 标准差
      stdDev: this.calculateStdDev(values)
    };
  }

  calculateStdDev(values) {
    const avg = values.reduce((a, b) => a + b, 0) / values.length;
    const squareDiffs = values.map(v => Math.pow(v - avg, 2));
    const avgSquareDiff = squareDiffs.reduce((a, b) => a + b, 0) / values.length;
    return Math.sqrt(avgSquareDiff);
  }
}
```

### 单向延迟 (OWD)

```javascript
// 需要时钟同步，实际难以精确测量
// 通常使用 RTT/2 估计

class OWDMeasurer {
  constructor() {
    this.clockOffset = 0;
  }

  // 估计时钟偏移
  estimateClockOffset(ntpTime, localTime) {
    // 简化：假设 RTT 对称
    this.clockOffset = ntpTime - localTime;
  }

  // 计算单向延迟
  calculateOWD(sendTime, receiveTime) {
    // OWD = 接收时间 - 发送时间 - 时钟偏移
    return receiveTime - sendTime - this.clockOffset;
  }
}
```

## 📦 丢包检测

### 丢包统计

```javascript
class PacketLossTracker {
  constructor() {
    this.expectedSeqNum = 0;
    this.receivedPackets = 0;
    this.lostPackets = 0;
    this.latePackets = 0;
    this.duplicatePackets = 0;

    // 历史记录
    this.lossHistory = [];
    this.historyWindow = 100; // 统计最近 100 个包
  }

  // 处理接收的包
  onPacketReceived(seqNum) {
    this.receivedPackets++;

    if (seqNum === this.expectedSeqNum) {
      // 正常顺序
      this.expectedSeqNum = (seqNum + 1) % 65536;
    } else if (this.isSeqNumAfter(seqNum, this.expectedSeqNum)) {
      // 跳过了一些包（可能丢失）
      const lost = this.seqNumDiff(seqNum, this.expectedSeqNum);
      this.lostPackets += lost;

      for (let i = 0; i < lost; i++) {
        this.lossHistory.push(true);
      }
      this.lossHistory.push(false);

      this.expectedSeqNum = (seqNum + 1) % 65536;
    } else {
      // 乱序包
      if (this.lossHistory[this.seqNumDiff(this.expectedSeqNum - 1, seqNum)]) {
        // 之前标记为丢失，现在收到了（迟到的包）
        this.latePackets++;
        this.lostPackets--;
      } else {
        // 重复包
        this.duplicatePackets++;
      }
    }

    // 限制历史长度
    while (this.lossHistory.length > this.historyWindow) {
      this.lossHistory.shift();
    }
  }

  // 计算丢包率
  getLossRate() {
    if (this.lossHistory.length === 0) return 0;
    const lost = this.lossHistory.filter(l => l).length;
    return lost / this.lossHistory.length;
  }

  // 序列号比较（处理回绕）
  isSeqNumAfter(a, b) {
    return ((a - b) & 0xFFFF) < 32768;
  }

  seqNumDiff(a, b) {
    return ((a - b) & 0xFFFF);
  }
}
```

### 突发丢包检测

```javascript
class BurstLossDetector {
  constructor() {
    this.recentLosses = [];
    this.windowSize = 50; // 统计窗口
    this.burstThreshold = 3; // 连续丢包阈值
  }

  recordPacket(lost) {
    this.recentLosses.push(lost);
    if (this.recentLosses.length > this.windowSize) {
      this.recentLosses.shift();
    }
  }

  // 检测突发丢包
  detectBurst() {
    let consecutiveLosses = 0;
    let maxConsecutive = 0;
    let burstCount = 0;

    for (const lost of this.recentLosses) {
      if (lost) {
        consecutiveLosses++;
        maxConsecutive = Math.max(maxConsecutive, consecutiveLosses);
      } else {
        if (consecutiveLosses >= this.burstThreshold) {
          burstCount++;
        }
        consecutiveLosses = 0;
      }
    }

    return {
      maxConsecutive,
      burstCount,
      isBursty: burstCount > 0 || maxConsecutive >= this.burstThreshold
    };
  }
}
```

## 📈 网络质量评估

### 综合评分

```javascript
class NetworkQualityAssessor {
  constructor() {
    this.weights = {
      bandwidth: 0.3,
      latency: 0.3,
      loss: 0.25,
      jitter: 0.15
    };
  }

  assess(stats) {
    const scores = {
      bandwidth: this.scoreBandwidth(stats.bandwidth),
      latency: this.scoreLatency(stats.rtt),
      loss: this.scoreLoss(stats.lossRate),
      jitter: this.scoreJitter(stats.jitter)
    };

    // 加权平均
    let totalScore = 0;
    for (const [key, weight] of Object.entries(this.weights)) {
      totalScore += scores[key] * weight;
    }

    return {
      overall: totalScore,
      details: scores,
      level: this.getQualityLevel(totalScore)
    };
  }

  scoreBandwidth(bandwidth) {
    // bps -> 0-100 分
    if (bandwidth >= 2000000) return 100;  // 2Mbps+
    if (bandwidth >= 1000000) return 80;   // 1Mbps+
    if (bandwidth >= 500000) return 60;    // 500kbps+
    if (bandwidth >= 200000) return 40;    // 200kbps+
    if (bandwidth >= 100000) return 20;    // 100kbps+
    return 10;
  }

  scoreLatency(rtt) {
    // seconds -> 0-100 分
    if (rtt <= 0.05) return 100;   // 50ms
    if (rtt <= 0.1) return 90;     // 100ms
    if (rtt <= 0.15) return 70;    // 150ms
    if (rtt <= 0.3) return 50;     // 300ms
    if (rtt <= 0.5) return 30;     // 500ms
    return 10;
  }

  scoreLoss(lossRate) {
    // 0-1 -> 0-100 分
    if (lossRate <= 0.01) return 100;  // 1%
    if (lossRate <= 0.02) return 90;   // 2%
    if (lossRate <= 0.05) return 70;   // 5%
    if (lossRate <= 0.1) return 50;    // 10%
    if (lossRate <= 0.2) return 30;    // 20%
    return 10;
  }

  scoreJitter(jitter) {
    // seconds -> 0-100 分
    if (jitter <= 0.01) return 100;  // 10ms
    if (jitter <= 0.03) return 90;   // 30ms
    if (jitter <= 0.05) return 70;   // 50ms
    if (jitter <= 0.1) return 50;    // 100ms
    return 30;
  }

  getQualityLevel(score) {
    if (score >= 80) return 'excellent';
    if (score >= 60) return 'good';
    if (score >= 40) return 'fair';
    if (score >= 20) return 'poor';
    return 'bad';
  }
}
```

## 🚨 网络告警

```javascript
class NetworkAlert {
  constructor() {
    this.thresholds = {
      highLatency: 0.2,      // 200ms
      highLoss: 0.05,        // 5%
      highJitter: 0.05,      // 50ms
      lowBandwidth: 100000   // 100kbps
    };

    this.alertCallbacks = [];
  }

  check(stats) {
    const alerts = [];

    if (stats.rtt > this.thresholds.highLatency) {
      alerts.push({
        type: 'high-latency',
        severity: 'warning',
        message: `高延迟: ${(stats.rtt * 1000).toFixed(0)}ms`
      });
    }

    if (stats.lossRate > this.thresholds.highLoss) {
      alerts.push({
        type: 'high-loss',
        severity: 'error',
        message: `高丢包: ${(stats.lossRate * 100).toFixed(1)}%`
      });
    }

    if (stats.jitter > this.thresholds.highJitter) {
      alerts.push({
        type: 'high-jitter',
        severity: 'warning',
        message: `高抖动: ${(stats.jitter * 1000).toFixed(0)}ms`
      });
    }

    if (stats.bandwidth < this.thresholds.lowBandwidth) {
      alerts.push({
        type: 'low-bandwidth',
        severity: 'error',
        message: `低带宽: ${(stats.bandwidth / 1000).toFixed(0)}kbps`
      });
    }

    // 触发回调
    alerts.forEach(alert => {
      this.alertCallbacks.forEach(cb => cb(alert));
    });

    return alerts;
  }

  onAlert(callback) {
    this.alertCallbacks.push(callback);
  }
}
```
