### 2. 模块化主进程

```javascript
// src/main/index.js
const { app, BrowserWindow } = require('electron')
const path = require('path')

// 导入模块
const { setupIpc } = require('./ipc')
const { setupStore } = require('./services/store')
const { setupTray } = require('./services/tray')
const { setupUpdate } = require('./services/update')

let mainWindow

async function createWindow() {
  mainWindow = new BrowserWindow({
    width: 1200,
    height: 800,
    webPreferences: {
      preload: path.join(__dirname, '../preload/index.js'),
      contextIsolation: true
    }
  })

  // 加载页面
  if (process.env.NODE_ENV === 'development') {
    await mainWindow.loadURL('http://localhost:5173')
  } else {
    await mainWindow.loadFile(path.join(__dirname, '../renderer/index.html'))
  }

  return mainWindow
}

async function bootstrap() {
  await app.whenReady()

  // 初始化服务
  setupStore()
  setupIpc()

  // 创建窗口
  const mainWindow = await createWindow()

  // 初始化需要窗口的服务
  setupTray(mainWindow)
  setupUpdate(mainWindow)

  // macOS 特殊处理
  app.on('activate', () => {
    if (BrowserWindow.getAllWindows().length === 0) {
      createWindow()
    }
  })
}

// 启动应用
bootstrap().catch(console.error)

// 退出处理
app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') {
    app.quit()
  }
})
```

## 🔄 IPC 架构

### 1. 模块化 IPC

```javascript
// src/main/ipc/index.js
const { ipcMain } = require('electron')

// 导入处理器
const fileHandlers = require('./file')
const windowHandlers = require('./window')
const settingsHandlers = require('./settings')

// 注册所有处理器
function setupIpc() {
  // 文件操作
  ipcMain.handle('file:read', fileHandlers.readFile)
  ipcMain.handle('file:write', fileHandlers.writeFile)
  ipcMain.handle('file:delete', fileHandlers.deleteFile)

  // 窗口操作
  ipcMain.handle('window:minimize', windowHandlers.minimize)
  ipcMain.handle('window:maximize', windowHandlers.maximize)
  ipcMain.handle('window:close', windowHandlers.close)

  // 设置操作
  ipcMain.handle('settings:get', settingsHandlers.get)
  ipcMain.handle('settings:set', settingsHandlers.set)

  console.log('IPC handlers registered')
}

module.exports = { setupIpc }
```

```javascript
// src/main/ipc/file.js
const fs = require('fs').promises
const path = require('path')

async function readFile(event, filePath) {
  try {
    const content = await fs.readFile(filePath, 'utf-8')
    return { success: true, content }
  } catch (error) {
    return { success: false, error: error.message }
  }
}

async function writeFile(event, filePath, content) {
  try {
    await fs.writeFile(filePath, content, 'utf-8')
    return { success: true }
  } catch (error) {
    return { success: false, error: error.message }
  }
}

async function deleteFile(event, filePath) {
  try {
    await fs.unlink(filePath)
    return { success: true }
  } catch (error) {
    return { success: false, error: error.message }
  }
}

module.exports = {
  readFile,
  writeFile,
  deleteFile
}
```

### 2. 类型安全的 IPC

```typescript
// src/shared/ipc-types.ts
export interface IpcChannels {
  'file:read': {
    request: { path: string }
    response: { success: boolean; content?: string; error?: string }
  }
  'file:write': {
    request: { path: string; content: string }
    response: { success: boolean; error?: string }
  }
  'settings:get': {
    request: { key: string }
    response: { value: unknown }
  }
}

// 类型安全的 IPC 调用
export type IpcHandler<T extends keyof IpcChannels> = (
  event: IpcMainInvokeEvent,
  ...args: Parameters<IpcChannels[T]['request']>
) => Promise<IpcChannels[T]['response']> | IpcChannels[T]['response']
```

## 📦 状态管理

### 1. 主进程状态

```javascript
// src/main/services/store.js
const Store = require('electron-store')
const { ipcMain } = require('electron')

const store = new Store({
  defaults: {
    window: {
      width: 1200,
      height: 800,
      x: undefined,
      y: undefined
    },
    settings: {
      theme: 'system',
      language: 'zh-CN',
      autoUpdate: true
    }
  }
})

function setupStore() {
  // 获取设置
  ipcMain.handle('store:get', (event, key) => {
    return store.get(key)
  })

  // 设置值
  ipcMain.handle('store:set', (event, key, value) => {
    store.set(key, value)
    return true
  })

  // 删除值
  ipcMain.handle('store:delete', (event, key) => {
    store.delete(key)
    return true
  })

  // 获取所有
  ipcMain.handle('store:getAll', () => {
    return store.store
  })
}

module.exports = { setupStore, store }
```

### 2. 渲染进程状态 (Pinia)

```javascript
// src/renderer/src/stores/app.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useAppStore = defineStore('app', () => {
  // 状态
  const theme = ref('system')
  const language = ref('zh-CN')
  const windowSize = ref({ width: 1200, height: 800 })

  // 计算属性
  const isDarkMode = computed(() => {
    if (theme.value === 'system') {
      return window.matchMedia('(prefers-color-scheme: dark)').matches
    }
    return theme.value === 'dark'
  })

  // 动作
  async function loadSettings() {
    const settings = await window.electronAPI.store.get('settings')
    if (settings) {
      theme.value = settings.theme
      language.value = settings.language
    }
  }

  async function saveSettings() {
    await window.electronAPI.store.set('settings', {
      theme: theme.value,
      language: language.value
    })
  }

  function setTheme(newTheme) {
    theme.value = newTheme
    saveSettings()
  }

  return {
    theme,
    language,
    windowSize,
    isDarkMode,
    loadSettings,
    saveSettings,
    setTheme
  }
})
```

### 3. 跨进程状态同步

```javascript
// 主进程
ipcMain.handle('state:sync', (event, state) => {
  // 更新主进程状态
  Object.assign(appState, state)

  // 广播给所有窗口
  BrowserWindow.getAllWindows().forEach(win => {
    win.webContents.send('state:updated', appState)
  })
})

// 渲染进程
const store = useAppStore()

// 监听主进程状态更新
window.electronAPI.onStateUpdated((state) => {
  store.$patch(state)
})

// 发送状态更新
watch(
  () => store.$state,
  (state) => {
    window.electronAPI.syncState(state)
  },
  { deep: true }
)
```

### 1. 服务抽象

```javascript
// src/main/services/base.js
class BaseService {
  constructor(name) {
    this.name = name
    this.initialized = false
  }

  async init() {
    if (this.initialized) return
    await this.setup()
    this.initialized = true
    console.log(`Service ${this.name} initialized`)
  }

  async setup() {
    throw new Error('setup() must be implemented')
  }

  async destroy() {
    this.initialized = false
    console.log(`Service ${this.name} destroyed`)
  }
}

module.exports = BaseService
```

### 2. 具体服务实现

```javascript
// src/main/services/database.js
const BaseService = require('./base')
const Database = require('better-sqlite3')
const path = require('path')

class DatabaseService extends BaseService {
  constructor() {
    super('database')
    this.db = null
  }

  async setup() {
    const dbPath = path.join(app.getPath('userData'), 'app.db')
    this.db = new Database(dbPath)

    // 初始化表
    this.db.exec(`
      CREATE TABLE IF NOT EXISTS notes (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        title TEXT NOT NULL,
        content TEXT,
        created_at DATETIME DEFAULT CURRENT_TIMESTAMP
      )
    `)
  }

  // CRUD 操作
  getNotes() {
    return this.db.prepare('SELECT * FROM notes ORDER BY created_at DESC').all()
  }

  createNote(title, content) {
    const stmt = this.db.prepare('INSERT INTO notes (title, content) VALUES (?, ?)')
    return stmt.run(title, content).lastInsertRowid
  }

  updateNote(id, title, content) {
    const stmt = this.db.prepare('UPDATE notes SET title = ?, content = ? WHERE id = ?')
    return stmt.run(title, content, id).changes
  }

  deleteNote(id) {
    const stmt = this.db.prepare('DELETE FROM notes WHERE id = ?')
    return stmt.run(id).changes
  }

  async destroy() {
    if (this.db) {
      this.db.close()
      this.db = null
    }
    await super.destroy()
  }
}

module.exports = new DatabaseService()
```

### 3. 服务管理器

```javascript
// src/main/services/index.js
const database = require('./database')
const store = require('./store')
const update = require('./update')
const tray = require('./tray')

class ServiceManager {
  constructor() {
    this.services = {
      database,
      store,
      update,
      tray
    }
  }

  async init() {
    for (const [name, service] of Object.entries(this.services)) {
      try {
        await service.init()
      } catch (error) {
        console.error(`Failed to initialize ${name}:`, error)
      }
    }
  }

  async destroy() {
    for (const [name, service] of Object.entries(this.services)) {
      try {
        await service.destroy()
      } catch (error) {
        console.error(`Failed to destroy ${name}:`, error)
      }
    }
  }

  get(name) {
    return this.services[name]
  }
}

module.exports = new ServiceManager()
```
