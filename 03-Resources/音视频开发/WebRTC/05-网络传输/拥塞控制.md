# 拥塞控制

> 拥塞控制是 WebRTC 自适应网络条件、保证通话质量的核心机制

## 📌 核心目标

- 估计可用带宽 (BWE)
- 避免网络拥塞
- 最大化利用带宽
- 保证低延迟

## 📊 带宽估计 (BWE)

### GCC (Google Congestion Control)

WebRTC 默认的拥塞控制算法，包含两部分：

1. **基于延迟的估计** (Delay-based)
2. **基于丢包的估计** (Loss-based)

### 基于延迟的估计

```javascript
// 原理：通过观察延迟变化判断拥塞

// 1. 计算单向延迟
function calculateOwd(arrivalTime, sendTime) {
  return arrivalTime - sendTime;
}

// 2. 计算延迟梯度
function calculateDelayGradient(owdSamples) {
  // 使用线性回归或卡尔曼滤波
  let gradient = 0;
  for (let i = 1; i < owdSamples.length; i++) {
    gradient += (owdSamples[i] - owdSamples[i - 1]);
  }
  return gradient / (owdSamples.length - 1);
}

// 3. 根据梯度调整带宽
function adjustBitrate(gradient, currentBitrate) {
  const THRESHOLD = 0; // 延迟梯度阈值

  if (gradient > THRESHOLD) {
    // 延迟增加，可能拥塞，降低码率
    return currentBitrate * 0.85;
  } else if (gradient < -THRESHOLD) {
    // 延迟减少，可以增加码率
    return currentBitrate * 1.03;
  } else {
    // 保持当前码率
    return currentBitrate;
  }
}
```

### 基于丢包的估计

```javascript
// 根据丢包率调整带宽

function adjustBitrateByLoss(lossRate, currentBitrate) {
  if (lossRate > 0.1) {
    // 丢包率 > 10%，大幅降低
    return currentBitrate * 0.5;
  } else if (lossRate > 0.02) {
    // 丢包率 2-10%，小幅降低
    return currentBitrate * 0.95;
  } else if (lossRate < 0.02) {
    // 丢包率 < 2%，可以增加
    return currentBitrate * 1.05;
  }
  return currentBitrate;
}
```

### 综合带宽估计

```javascript
class BandwidthEstimator {
  constructor() {
    this.delayBasedBitrate = 0;
    this.lossBasedBitrate = 0;
    this.currentBitrate = 300000; // 初始 300kbps
    this.minBitrate = 30000;      // 最小 30kbps
    this.maxBitrate = 5000000;    // 最大 5Mbps
  }

  update(delayGradient, lossRate) {
    // 基于延迟的估计
    this.delayBasedBitrate = this.estimateFromDelay(delayGradient);

    // 基于丢包的估计
    this.lossBasedBitrate = this.estimateFromLoss(lossRate);

    // 取较小值（更保守）
    this.currentBitrate = Math.min(
      this.delayBasedBitrate,
      this.lossBasedBitrate
    );

    // 限制范围
    this.currentBitrate = Math.max(
      this.minBitrate,
      Math.min(this.maxBitrate, this.currentBitrate)
    );

    return this.currentBitrate;
  }

  estimateFromDelay(gradient) {
    // Trendline 滤波器实现
    if (gradient > 0) {
      return this.currentBitrate * 0.85;
    } else {
      return this.currentBitrate * 1.03;
    }
  }

  estimateFromLoss(lossRate) {
    if (lossRate > 0.1) {
      return this.currentBitrate * 0.5;
    } else if (lossRate > 0.02) {
      return this.currentBitrate * 0.95;
    }
    return this.currentBitrate * 1.05;
  }
}
```

## 📡 Transport-CC

### 工作原理

```
发送端                                      接收端
   │                                           │
   │  RTP 包 (带 Transport-CC 扩展头)          │
   │  包含发送时间戳                           │
   │ ────────────────────────────────────────► │
   │                                           │
   │                          记录接收时间
   │                          计算延迟
   │                                           │
   │  RTCP Transport-CC Feedback               │
   │  包含所有包的接收状态和延迟                │
   │ ◄───────────────────────────────────────  │
   │                                           │
   │  根据反馈估计带宽
   │  调整发送码率
   │                                           │
```

### Transport-CC 扩展头

// RTP 扩展头 (1字节)
// 0 1 2 3 4 5 6 7

// +-+-+-+-+-+-+-+-+
// |   Transport Wide Sequence Number (16 bits)  |
// |                                               |
// +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+

a=extmap:3 http://www.ietf.org/id/draft-holmer-rmcat-transport-wide-cc-extensions-01

### 接收端反馈

```javascript
class TransportCCReceiver {
  constructor() {
    this.packets = new Map();
    this.feedbackInterval = 100; // ms
  }

  // 记录接收的包
  onPacketReceived(seqNum, sendTime, arrivalTime) {
    this.packets.set(seqNum, {
      sendTime,
      arrivalTime,
      received: true
    });
  }

  // 生成反馈
  generateFeedback() {
    const feedback = {
      mediaSsrc: 0,
      baseSeqNum: 0,
      packetStatusCount: 0,
      referenceTime: 0,
      fbPktCount: 0,
      packetChunks: [],
      recvDeltas: []
    };

    // 遍历所有包，生成状态
    this.packets.forEach((packet, seqNum) => {
      feedback.packetChunks.push({
        seqNum,
        received: packet.received,
        delay: packet.arrivalTime - packet.sendTime
      });
    });

    this.packets.clear();
    return feedback;
  }
}
```

## 🎛️ 码率控制

### 码率分配

```javascript
class BitrateAllocator {
  constructor() {
    this.totalBitrate = 0;
    this.videoBitrate = 0;
    this.audioBitrate = 0;
    this.fecBitrate = 0;
    this.retransmissionBitrate = 0;
  }

  allocate(totalBitrate) {
    this.totalBitrate = totalBitrate;

    // 优先保证音频
    this.audioBitrate = Math.min(32000, totalBitrate * 0.1);

    // 剩余给视频
    const remaining = totalBitrate - this.audioBitrate;

    // FEC 和重传占用部分带宽
    this.fecBitrate = remaining * 0.1;
    this.retransmissionBitrate = remaining * 0.1;

    // 视频使用剩余
    this.videoBitrate = remaining - this.fecBitrate - this.retransmissionBitrate;

    return {
      audio: this.audioBitrate,
      video: this.videoBitrate,
      fec: this.fecBitrate,
      retransmission: this.retransmissionBitrate
    };
  }
}
```

### 应用码率到编码器

```javascript
async function applyBitrate(pc, bitrate) {
  const senders = pc.getSenders();

  for (const sender of senders) {
    if (sender.track?.kind === 'video') {
      const parameters = sender.getParameters();

      if (!parameters.encodings) {
        parameters.encodings = [{}];
      }

      parameters.encodings[0].maxBitrate = bitrate.video;
      parameters.encodings[0].maxFramerate = calculateFramerate(bitrate.video);

      await sender.setParameters(parameters);
    }
  }
}

function calculateFramerate(bitrate) {
  // 根据码率调整帧率
  if (bitrate < 150000) return 15;
  if (bitrate < 300000) return 24;
  return 30;
}
```

## 📈 性能监控

### 获取带宽估计

```javascript
async function getBandwidthEstimate(pc) {
  const stats = await pc.getStats();
  let estimate = {
    availableOutgoingBitrate: 0,
    availableIncomingBitrate: 0,
    currentRoundTripTime: 0
  };

  stats.forEach(report => {
    if (report.type === 'candidate-pair' && report.state === 'succeeded') {
      estimate.availableOutgoingBitrate = report.availableOutgoingBitrate;
      estimate.availableIncomingBitrate = report.availableIncomingBitrate;
      estimate.currentRoundTripTime = report.currentRoundTripTime;
    }
  });

  return estimate;
}

// 定期监控
setInterval(async () => {
  const estimate = await getBandwidthEstimate(pc);
  console.log('带宽估计:', {
    outgoing: `${(estimate.availableOutgoingBitrate / 1000000).toFixed(2)} Mbps`,
    incoming: `${(estimate.availableIncomingBitrate / 1000000).toFixed(2)} Mbps`,
    rtt: `${(estimate.currentRoundTripTime * 1000).toFixed(0)} ms`
  });
}, 1000);
```

### 拥塞信号

```javascript
function detectCongestion(stats) {
  const signals = {
    highLoss: stats.packetsLost / stats.packetsReceived > 0.05,
    highJitter: stats.jitter > 0.05,
    highRtt: stats.roundTripTime > 0.2,
    lowBitrate: stats.bitrate < 100000
  };

  const isCongested = Object.values(signals).some(v => v);

  return {
    isCongested,
    signals
  };
}
```
